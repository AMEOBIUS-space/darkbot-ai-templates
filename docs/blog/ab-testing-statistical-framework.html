The Statistics (Simplified)

>

You don't need a PhD. You need three things:

>
  1. Sample size— how many observations to detect a real difference
  2. p-value— probability the difference is noise (want < 0.05)
  3. Confidence interval— range where the true difference lies
>

Minimum Sample Size Calculator

>
import math

def min_sample_size(baseline_rate: float,
 min_detectable_effect: float,
 alpha: float = 0.05,
 power: float = 0.80) ->int:
 """
 How many samples per variant to detect a difference.
 
 baseline_rate: Expected success rate of control (0.0-1.0)
 min_detectable_effect: Smallest change you care about (0.0-1.0)
 alpha: False positive rate (5% = 0.05)
 power: Probability of detecting a real effect (80% = 0.80)
 """
 # Z-scores for common alpha/power
 z_alpha = {0.05: 1.96, 0.01: 2.576}
 z_power = {0.80: 0.84, 0.90: 1.28, 0.95: 1.645}

 za = z_alpha.get(alpha, 1.96)
 zb = z_power.get(power, 0.84)

 p1 = baseline_rate
 p2 = baseline_rate + min_detectable_effect
 p_avg = (p1 + p2) / 2

 numerator = (za * math.sqrt(2 * p_avg * (1 - p_avg))
 + zb * math.sqrt(p1 * (1 - p1) + p2 * (1 - p2))) ** 2
 denominator = (p2 - p1) ** 2

 return math.ceil(numerator / denominator)


# How many samples to detect a 5% accuracy improvement from 70% baseline?
n = min_sample_size(baseline_rate=0.70, min_detectable_effect=0.05)
print(f"Need {n} samples per variant")

# Output: Need 823 samples per variant
>
Rule of thumb:For a 5% accuracy improvement at 70% baseline, you need ~800 samples per variant. For a 10% improvement, ~200. For a 20% improvement, ~50. Bigger effects are easier to detect.

The Experiment Framework

>
import json
import random
import time
import sqlite3
from dataclasses import dataclass, field
from typing import Any

@dataclass
class ExperimentConfig:
 """One side of an A/B test."""
 name: str
 model: str = ""
 system_prompt: str = ""
 temperature: float = 0.0
 enabled_tools: list[str] = field(default_factory=list)
 metadata: dict[str, Any] = field(default_factory=dict)


@dataclass
class ExperimentResult:
 """One observation from the experiment."""
 experiment_id: str
 variant: str # "A" or "B"
 success: bool # Did the agent complete the task?
 accuracy: float # 0.0-1.0 quality score
 tokens_used: int
 latency_ms: float
 cost_usd: float
 timestamp: float = field(default_factory=time.time)
 metadata: dict[str, Any] = field(default_factory=dict)


class ABTestFramework:
 def __init__(self, db_path: str = ":memory:"):
 self.conn = sqlite3.connect(db_path)
 self.conn.execute("""
 CREATE TABLE IF NOT EXISTS experiments (
 experiment_id TEXT,
 variant TEXT,
 config TEXT,
 created_at REAL,
 status TEXT DEFAULT 'running',
 PRIMARY KEY (experiment_id, variant)
 )
 """)
 self.conn.execute("""
 CREATE TABLE IF NOT EXISTS observations (
 id INTEGER PRIMARY KEY AUTOINCREMENT,
 experiment_id TEXT,
 variant TEXT,
 success INTEGER,
 accuracy REAL,
 tokens_used INTEGER,
 latency_ms REAL,
 cost_usd REAL,
 timestamp REAL,
 metadata TEXT
 )
 """)
 self.conn.commit()

 def create_experiment(self, experiment_id: str,
 config_a: ExperimentConfig,
 config_b: ExperimentConfig) ->None:
 for variant, config in [("A", config_a), ("B", config_b)]:
 self.conn.execute("""
 INSERT INTO experiments (experiment_id, variant, config, created_at)
 VALUES (?, ?, ?, ?)
 """, (experiment_id, variant, json.dumps(config.__dict__),
 time.time()))
 self.conn.commit()

 def assign_variant(self, experiment_id: str,
 user_id: str | None = None) ->str:
 """Deterministic variant assignment — same user always gets same variant."""
 if user_id:
 # Hash-based assignment for consistency
 h = hash(f"{experiment_id}:{user_id}") % 2
 return "A" if h == 0 else "B"
 else:
 return random.choice(["A", "B"])

 def record(self, result: ExperimentResult) ->None:
 self.conn.execute("""
 INSERT INTO observations
 (experiment_id, variant, success, accuracy, tokens_used,
 latency_ms, cost_usd, timestamp, metadata)
 VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
 """, (
 result.experiment_id, result.variant,
 int(result.success), result.accuracy,
 result.tokens_used, result.latency_ms,
 result.cost_usd, result.timestamp,
 json.dumps(result.metadata),
 ))
 self.conn.commit()

Statistical Analysis

>
import math

class ABAnalyzer:
 """Statistical analysis of A/B test results."""

 def __init__(self, framework: ABTestFramework):
 self.fw = framework

 def get_results(self, experiment_id: str) ->dict:
 """Aggregate results for both variants."""

 results = {}
 for variant in ["A", "B"]:
 rows = self.fw.conn.execute("""
 SELECT success, accuracy, tokens_used, latency_ms, cost_usd
 FROM observations
 WHERE experiment_id = ? AND variant = ?
 """, (experiment_id, variant)).fetchall()

 if not rows:
 results[variant] = {"samples": 0}
 continue

 n = len(rows)
 successes = sum(1 for r in rows if r[0])
 results[variant] = {
 "samples": n,
 "success_rate": successes / n,
 "accuracy_mean": sum(r[1] for r in rows) / n,
 "accuracy_std": self._std([r[1] for r in rows]),
 "tokens_mean": sum(r[2] for r in rows) / n,
 "latency_mean": sum(r[3] for r in rows) / n,
 "cost_mean": sum(r[4] for r in rows) / n,
 }

 return results

 def significance_test(self, experiment_id: str,
 metric: str = "accuracy") ->dict:
 """
 Two-proportion z-test for success rates.
 Two-sample t-test for continuous metrics.
 """
 results = self.get_results(experiment_id)

 a = results.get("A", {})
 b = results.get("B", {})

 if a.get("samples", 0) < 2 or b.get("samples", 0) < 2:
 return {"status": "insufficient_data",
 "samples_a": a.get("samples", 0),
 "samples_b": b.get("samples", 0)}

 if metric == "success_rate":
 return self._proportion_test(a, b)
 else:
 return self._continuous_test(a, b, metric)

 def _proportion_test(self, a: dict, b: dict) ->dict:
 """Z-test for difference in proportions (success rate)."""
 n_a, n_b = a["samples"], b["samples"]
 p_a, p_b = a["success_rate"], b["success_rate"]

 # Pooled proportion
 p_pooled = (p_a * n_a + p_b * n_b) / (n_a + n_b)
 se = math.sqrt(p_pooled * (1 - p_pooled) * (1/n_a + 1/n_b))

 if se == 0:
 z_score = 0
 else:
 z_score = (p_b - p_a) / se

 p_value = 2 * (1 - self._normal_cdf(abs(z_score)))

 # Confidence interval
 diff = p_b - p_a
 se_unpooled = math.sqrt(
 p_a * (1 - p_a) / n_a + p_b * (1 - p_b) / n_b
 )
 ci_low = diff - 1.96 * se_unpooled
 ci_high = diff + 1.96 * se_unpooled

 return {
 "metric": "success_rate",
 "variant_a": round(p_a, 4),
 "variant_b": round(p_b, 4),
 "difference": round(diff, 4),
 "z_score": round(z_score, 3),
 "p_value": round(p_value, 4),
 "ci_95": [round(ci_low, 4), round(ci_high, 4)],
 "significant": p_value < 0.05,
 "winner": self._declare_winner(diff, p_value),
 }

 def _continuous_test(self, a: dict, b: dict,
 metric: str) ->dict:
 """Welch's t-test for continuous metrics (accuracy, tokens, etc.)."""
 n_a, n_b = a["samples"], b["samples"]
 m_a = a.get(f"{metric}_mean", 0)
 m_b = b.get(f"{metric}_mean", 0)
 s_a = a.get(f"{metric}_std", 0) or 0.001
 s_b = b.get(f"{metric}_std", 0) or 0.001

 # Welch's t-statistic
 se = math.sqrt(s_a**2 / n_a + s_b**2 / n_b)
 t_score = (m_b - m_a) / se if se >0 else 0

 # Welch-Satterthwaite degrees of freedom
 num = (s_a**2 / n_a + s_b**2 / n_b) ** 2
 den = (s_a**2 / n_a) ** 2 / (n_a - 1) + (s_b**2 / n_b) ** 2 / (n_b - 1)
 df = num / den if den >0 else n_a + n_b - 2

 # Approximate p-value using normal distribution
 # (good approximation for df >30)
 p_value = 2 * (1 - self._normal_cdf(abs(t_score)))

 diff = m_b - m_a
 ci_low = diff - 1.96 * se
 ci_high = diff + 1.96 * se

 return {
 "metric": metric,
 "variant_a": round(m_a, 4),
 "variant_b": round(m_b, 4),
 "difference": round(diff, 4),
 "t_score": round(t_score, 3),
 "degrees_of_freedom": round(df, 1),
 "p_value": round(p_value, 4),
 "ci_95": [round(ci_low, 4), round(ci_high, 4)],
 "significant": p_value < 0.05,
 "winner": self._declare_winner(diff, p_value),
 }

 def _declare_winner(self, diff: float, p_value: float) ->str:
 if p_value >= 0.05:
 return "no_significant_difference"
 return "B" if diff >0 else "A"

 def _normal_cdf(self, z: float) ->float:
 """Approximate normal CDF using error function."""
 return 0.5 * (1 + math.erf(z / math.sqrt(2)))

 def _std(self, values: list[float]) ->float:
 n = len(values)
 if n < 2:
 return 0.0
 mean = sum(values) / n
 variance = sum((v - mean) ** 2 for v in values) / (n - 1)
 return math.sqrt(variance)

The Full Experiment

>
# --- Setup ---
fw = ABTestFramework("ab_tests.db")
analyzer = ABAnalyzer(fw)

# --- Experiment: GPT-4o vs Claude Sonnet ---
exp_id = "model-compare-001"

config_a = ExperimentConfig(
 name="gpt-4o",
 model="gpt-4o",
 temperature=0.0,
 system_prompt="You are a coding assistant. Answer concisely.",
)

config_b = ExperimentConfig(
 name="claude-sonnet",
 model="claude-sonnet-4-20250514",
 temperature=0.0,
 system_prompt="You are a coding assistant. Answer concisely.",
)

fw.create_experiment(exp_id, config_a, config_b)

# --- Simulate 200 observations per variant ---
import random
random.seed(42)

for i in range(200):
 variant = fw.assign_variant(exp_id, user_id=f"user_{i}")

 if variant == "A":
 # GPT-4o: 78% accuracy, ~450 tokens, ~1.2s
 accuracy = random.gauss(0.78, 0.12)
 success = accuracy >0.6
 tokens = random.gauss(450, 80)
 latency = random.gauss(1200, 200)
 cost = tokens * 0.000005 # $5/1M tokens
 else:
 # Claude: 82% accuracy, ~380 tokens, ~1.5s
 accuracy = random.gauss(0.82, 0.10)
 success = accuracy >0.6
 tokens = random.gauss(380, 60)
 latency = random.gauss(1500, 300)
 cost = tokens * 0.000003 # $3/1M tokens

 fw.record(ExperimentResult(
 experiment_id=exp_id,
 variant=variant,
 success=success,
 accuracy=max(0, min(1, accuracy)),
 tokens_used=int(tokens),
 latency_ms=latency,
 cost_usd=cost,
 ))

# --- Analyze ---
print("=== Success Rate ===")
sr = analyzer.significance_test(exp_id, metric="success_rate")
for k, v in sr.items():
 print(f" {k}: {v}")

print("\n=== Accuracy ===")
acc = analyzer.significance_test(exp_id, metric="accuracy")
for k, v in acc.items():
 print(f" {k}: {v}")

print("\n=== Token Usage ===")
tok = analyzer.significance_test(exp_id, metric="tokens")
for k, v in tok.items():
 print(f" {k}: {v}")

print("\n=== Cost ===")
cost_result = analyzer.get_results(exp_id)
print(f" A avg cost: ${cost_result['A']['cost_mean']:.4f}")
print(f" B avg cost: ${cost_result['B']['cost_mean']:.4f}")

Decision Framework

>
ScenarioAction
Accuracy ↑, Cost ↓Ship B immediately
Accuracy ↑, Cost ↑Calculate ROI — is accuracy gain worth the cost?
Accuracy =, Cost ↓Ship B — free cost savings
Accuracy =, Cost =No action — results are noise
Accuracy ↓Keep A, investigate B's failure mode

Cost-Adjusted Accuracy

>

When accuracy gains come at higher cost, compute cost-adjusted accuracy:

>
def cost_adjusted_score(accuracy: float, cost_usd: float,
 budget_per_call: float = 0.01) ->float:
 """Penalize accuracy by cost overrun."""
 if cost_usd <= budget_per_call:
 return accuracy
 penalty = (cost_usd - budget_per_call) / budget_per_call
 return accuracy * (1 - penalty * 0.5) # 50% penalty weight

score_a = cost_adjusted_score(0.78, 0.00225, budget=0.005)
score_b = cost_adjusted_score(0.82, 0.00114, budget=0.005)
print(f"Cost-adjusted: A={score_a:.3f}, B={score_b:.3f}")
# Cost-adjusted: A=0.780, B=0.820 — B wins even without cost penalty
>

Common Pitfalls

>
Pitfall 1: Peeking at results.Checking p-values every 10 samples and stopping when significant inflates false positives. Decide your sample size upfront and stick to it.
>
Pitfall 2: Multiple comparisons.If you test 5 metrics, one will be "significant" by chance. Apply Bonferroni correction: divide alpha (0.05) by the number of metrics tested.
>
Pitfall 3: Simpson's paradox.Variant B might win overall but lose in every subgroup. Always check segment-level results (by task type, user type, difficulty).
>
Pitfall 4: Non-stationary effects.Agent performance changes over time as models update. Don't run experiments for more than 2 weeks — external changes confound results.
>

Conclusion

>

A/B testing AI agents is simpler than it looks. The math is 200 years old. The framework is 300 lines of Python. The discipline is what matters:

>
  1. Decide sample size before you start.
  2. Test one variable at a time.
  3. Wait for significance before declaring a winner.
  4. Check segments, not just aggregates.
>

Stop arguing about which model "feels better." Run the test. Let the numbers decide.
