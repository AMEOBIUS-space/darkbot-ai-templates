Permission Systems for AI Agents: Allow/Deny Rules

>

SecurityMCPPython

>

2026-07-09

>

Your AI agent has a terminal. It can run any command. What stops it from running rm -rf /or git push --force?

>

Approval prompts are one answer, but they interrupt flow and create alert fatigue. A permission system with rules is better: the agent operates freely within guardrails, and you review the audit log later.

>

Rule-Based Permission Guard

>
import fnmatch
import re
from dataclasses import dataclass, field
from enum import Enum
from typing import Optional

class Decision(Enum):
 ALLOW = "allow"
 DENY = "deny"
 PROMPT = "prompt" # Ask the user

@dataclass
class Rule:
 name: str
 decision: Decision
 tool: str = "*" # Tool name pattern (glob)
 args: dict = field(default_factory=dict) # Argument matchers
 reason: str = ""
 priority: int = 0 # Higher = checked first

@dataclass
class PermissionGuard:
 rules: list = field(default_factory=list)
 audit_log: list = field(default_factory=list)
 
 def add_rule(self, rule: Rule):
 self.rules.append(rule)
 self.rules.sort(key=lambda r: r.priority, reverse=True)
 
 def check(self, tool_name: str, args: dict) ->dict:
 """Check if a tool call is permitted."""
 for rule in self.rules:
 if self._matches(rule, tool_name, args):
 decision = rule.decision
 self._log(tool_name, args, decision, rule.name)
 return {
 "decision": decision.value,
 "rule": rule.name,
 "reason": rule.reason,
 "tool": tool_name,
 "args": args
 }
 
 # Default: prompt for unknown
 self._log(tool_name, args, Decision.PROMPT, "default")
 return {
 "decision": Decision.PROMPT.value,
 "rule": "default",
 "reason": "No matching rule",
 "tool": tool_name,
 "args": args
 }
 
 def _matches(self, rule: Rule, tool_name: str, args: dict) ->bool:
 """Check if a rule matches the given tool call."""
 # Tool name match (glob)
 if not fnmatch.fnmatch(tool_name, rule.tool):
 return False
 
 # Argument matchers
 for key, matcher in rule.args.items():
 if key not in args:
 return False
 
 value = str(args[key])
 
 if isinstance(matcher, str):
 # Glob match for strings
 if matcher.startswith("regex:"):
 pattern = matcher[6:]
 if not re.search(pattern, value):
 return False
 elif not fnmatch.fnmatch(value, matcher):
 return False
 elif callable(matcher):
 if not matcher(value):
 return False
 
 return True
 
 def _log(self, tool: str, args: dict, decision: Decision, rule: str):
 from datetime import datetime
 self.audit_log.append({
 "timestamp": datetime.now().isoformat(),
 "tool": tool,
 "args": str(args)[:200], # truncate
 "decision": decision.value,
 "rule": rule
 })

Rule Patterns

>

Pattern 1: Safe Read Operations (Always Allow)

>
>
guard.add_rule(Rule(
 name="allow_reads",
 decision=Decision.ALLOW,
 tool="read_file",
 priority=100,
 reason="File reads are safe"
))

guard.add_rule(Rule(
 name="allow_safe_terminal",
 decision=Decision.ALLOW,
 tool="terminal",
 args={"command": "ls*"}, # Only ls commands
 priority=100,
 reason="Directory listing is safe"
))

guard.add_rule(Rule(
 name="allow_git_status",
 decision=Decision.ALLOW,
 tool="terminal",
 args={"command": "git status*"},
 priority=100,
 reason="Git status is read-only"
))
>
>

Pattern 2: Destructive Operations (Deny)

>
>
guard.add_rule(Rule(
 name="deny_rm_rf",
 decision=Decision.DENY,
 tool="terminal",
 args={"command": "regex:.*rm\\s+-rf.*"},
 priority=200,
 reason="Recursive force delete is blocked"
))

guard.add_rule(Rule(
 name="deny_force_push",
 decision=Decision.DENY,
 tool="terminal",
 args={"command": "regex:.*git\\s+push.*--force.*"},
 priority=200,
 reason="Force push can destroy history"
))

guard.add_rule(Rule(
 name="deny_env_access",
 decision=Decision.DENY,
 tool="read_file",
 args={"path": "regex:.*\\.env.*"},
 priority=200,
 reason="Environment files contain secrets"
))
>
>

Pattern 3: Network Operations (Prompt)

>
guard.add_rule(Rule(
 name="prompt_network",
 decision=Decision.PROMPT,
 tool="terminal",
 args={"command": "regex:.*(curl|wget|ssh|scp).*"},
 priority=150,
 reason="Network operations require approval"
))

guard.add_rule(Rule(
 name="prompt_package_install",
 decision=Decision.PROMPT,
 tool="terminal",
 args={"command": "regex:.*(pip\\s+install|npm\\s+install|apt).*"},
 priority=150,
 reason="Package installation requires approval"
))
>

Pattern 4: Working Directory Restriction

>
guard.add_rule(Rule(
 name="deny_outside_project",
 decision=Decision.DENY,
 tool="terminal",
 args={"command": "regex:.*\\s+/(?!home/user/project).*"},
 priority=180,
 reason="Operations outside project directory blocked"
))

# More specific: block writes to /etc, /usr, /boot
guard.add_rule(Rule(
 name="deny_system_dirs",
 decision=Decision.DENY,
 tool="terminal",
 args={"command": "regex:.*(>/|\\|tee\\s+)/(etc|usr|boot|sys|proc).*"},
 priority=190,
 reason="System directory access blocked"
))
>

Pattern 5: Resource Limits

>
guard.add_rule(Rule(
 name="deny_large_writes",
 decision=Decision.DENY,
 tool="write_file",
 args={"content": lambda v: len(v) >1_000_000}, # 1MB limit
 priority=170,
 reason="Files over 1MB require manual creation"
))

Combining Rules into a Policy

>
guard = PermissionGuard()

# === DENY RULES (highest priority) ===
guard.add_rule(Rule("deny_rm_rf", Decision.DENY, "terminal", 
 {"command": "regex:.*rm\\s+-rf.*"}, "Blocked: destructive", 200))
guard.add_rule(Rule("deny_force_push", Decision.DENY, "terminal",
 {"command": "regex:.*git\\s+push.*--force.*"}, "Blocked: history loss", 200))
guard.add_rule(Rule("deny_env", Decision.DENY, "read_file",
 {"path": "regex:.*\\.env.*"}, "Blocked: secrets", 200))
guard.add_rule(Rule("deny_sudo", Decision.DENY, "terminal",
 {"command": "regex:.*sudo.*"}, "Blocked: privilege escalation", 200))

# === PROMPT RULES (medium priority) ===
guard.add_rule(Rule("prompt_network", Decision.PROMPT, "terminal",
 {"command": "regex:.*(curl|wget|http).*"}, "Network needs approval", 150))
guard.add_rule(Rule("prompt_install", Decision.PROMPT, "terminal",
 {"command": "regex:.*(pip|npm|apt).*"}, "Installs need approval", 150))

# === ALLOW RULES (lower priority) ===
guard.add_rule(Rule("allow_reads", Decision.ALLOW, "read_file", priority=100))
guard.add_rule(Rule("allow_search", Decision.ALLOW, "search_files", priority=100))
guard.add_rule(Rule("allow_ls", Decision.ALLOW, "terminal",
 {"command": "ls*"}, "Listing is safe", 100))
guard.add_rule(Rule("allow_git_read", Decision.ALLOW, "terminal",
 {"command": "regex:git\\s+(status|log|diff|branch).*"}, 
 "Git reads are safe", 100))

# Default: prompt for anything unmatched
# (handled by the guard's default behavior)

Integration Example

>
# In your agent loop, before executing any tool:
def execute_tool(tool_name, args, guard):
 permission = guard.check(tool_name, args)
 
 if permission["decision"] == "deny":
 return f"[BLOCKED] {permission['reason']} (rule: {permission['rule']})"
 
 if permission["decision"] == "prompt":
 approved = ask_user(
 f"Allow {tool_name}({args})? Reason: {permission['reason']}"
 )
 if not approved:
 return "[CANCELLED] User denied tool call"
 
 # Execute the tool
 result = actual_tool_handler(tool_name, args)
 return result

# The guard automatically logs every decision
# Review the audit log:
for entry in guard.audit_log[-20:]:
 print(f"{entry['timestamp']} | {entry['decision']:8} | "
 f"{entry['tool']:15} | {entry['rule']}")

Advanced: Conditional Rules

>
# Allow git push only to feature branches
guard.add_rule(Rule(
 name="allow_feature_push",
 decision=Decision.ALLOW,
 tool="terminal",
 args={
 "command": "regex:git\\s+push\\s+origin\\s+feat/.*"
 },
 priority=160,
 reason="Feature branch pushes are safe"
))

# Deny pushes to main/master
guard.add_rule(Rule(
 name="deny_main_push",
 decision=Decision.DENY,
 tool="terminal",
 args={
 "command": "regex:git\\s+push\\s+origin\\s+(main|master).*"
 },
 priority=210, # Higher than feature push
 reason="Direct pushes to main are blocked"
))
