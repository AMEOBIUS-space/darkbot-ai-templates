Agent Observability: Three Layers Every AI Harness Needs

>

AI AgentsObservabilityPythonMCP

>

You shipped an AI agent. It works on your machine. In production, it calls the wrong tools, loops on failures, and burns through your API budget in forty minutes. Where did it go wrong?

>

Traditional APM (Datadog, New Relic) wasn't built for agents. They trace HTTP requests. Agents trace reasoning loops— model calls that spawn tool calls that spawn more model calls, sometimes ten layers deep. You need a different architecture.

>

This guide covers the three observability layers that matter for AI agents, with latency numbers and code from our harness engineering MCP suite.

>

Layer 1: Distributed Tracing (The "What Happened" Layer)

>

Every agent turn produces a tree: model call → tool calls → sub-model calls. Each node needs a span with timing, input summary, and output summary.

>

Minimal Span Implementation

>

Using Python's contextvarsto thread trace context through async calls — zero external dependencies:

>
import contextvars
import json
import time
import uuid
from dataclasses import dataclass, field
from typing import Any

# Context variable holds the current span
_current_span: contextvars.ContextVar["Span"] = contextvars.ContextVar("current_span")

@dataclass
class Span:
 name: str
 span_id: str = field(default_factory=lambda: uuid.uuid4().hex[:16])
 parent_id: str | None = None
 start_time: float = field(default_factory=time.monotonic)
 end_time: float | None = None
 attributes: dict[str, Any] = field(default_factory=dict)
 events: list[dict] = field(default_factory=list)

 def set_attribute(self, key: str, value: Any) ->None:
 # Serialize non-primitive values
 if isinstance(value, (dict, list)):
 self.attributes[key] = json.dumps(value)[:500]
 else:
 self.attributes[key] = value

 def add_event(self, name: str, **kwargs) ->None:
 self.events.append({
 "name": name,
 "timestamp": time.monotonic(),
 "attributes": kwargs,
 })

 def finish(self) ->None:
 self.end_time = time.monotonic()

 @property
 def duration_ms(self) ->float:
 end = self.end_time or time.monotonic()
 return (end - self.start_time) * 1000


class AgentTracer:
 """Minimal tracer — no OpenTelemetry, no protobuf, just JSON."""

 def __init__(self):
 self._spans: list[Span] = []

 def start_span(self, name: str, **attributes) ->Span:
 parent = _current_span.get(None)
 span = Span(name=name, parent_id=parent.span_id if parent else None)

 for k, v in attributes.items():
 span.set_attribute(k, v)

 self._spans.append(span)

 # Set as current span in context
 token = _current_span.set(span)
 span._token = token # Store for cleanup
 return span

 def end_span(self, span: Span) ->None:
 span.finish()
 if hasattr(span, "_token"):
 _current_span.reset(span._token)

 def export(self) ->str:
 """Export all spans as JSON — ready for any viewer."""
 return json.dumps([{
 "span_id": s.span_id,
 "parent_id": s.parent_id,
 "name": s.name,
 "duration_ms": round(s.duration_ms, 2),
 "attributes": s.attributes,
 "events": s.events,
 } for s in self._spans], indent=2)


# Usage in agent loop
tracer = AgentTracer()

def run_turn(user_message: str) ->str:
 with tracer_span("agent_turn", user_msg=user_message[:200]):
 response = call_model(user_message)

 if response.tool_calls:
 for tc in response.tool_calls:
 with tracer_span("tool_call",
 tool_name=tc.name,
 args_summary=str(tc.args)[:200]):
 result = execute_tool(tc.name, tc.args)
 # Add result as event

 return response.content


from contextlib import contextmanager

@contextmanager
def tracer_span(name: str, **attrs):
 span = tracer.start_span(name, **attrs)
 try:
 yield span
 except Exception as e:
 span.add_event("error", type=type(e).__name__, message=str(e))
 raise
 finally:
 tracer.end_span(span)
>

Layer 2: Metrics Aggregation (The "How Bad" Layer)

>

Traces tell you about one turn. Metrics tell you about patterns across thousands of turns. Four metric types cover 95% of agent observability:

>

1. Counter — Total Events

>
class Counter:
 def __init__(self, name: str):
 self.name = name
 self._value = 0
 self._labels: dict[str, int] = {} # label_string → count

 def inc(self, n: int = 1, **labels) ->None:
 self._value += n
 if labels:
 key = json.dumps(labels, sort_keys=True)
 self._labels[key] = self._labels.get(key, 0) + n

 @property
 def value(self) ->int:
 return self._value

 def by_label(self) ->dict[str, int]:
 return dict(self._labels)

# Track tool usage
tool_calls = Counter("agent_tool_calls_total")
tool_calls.inc(tool_name="read_file")
tool_calls.inc(tool_name="read_file")
tool_calls.inc(tool_name="search_web")

print(tool_calls.by_label())
# {'{"tool_name": "read_file"}': 2, '{"tool_name": "search_web"}': 1}
>

2. Histogram — Latency Distribution

>
class Histogram:
 """Fixed-bucket histogram — no floating point needed."""

 BUCKETS = [0, 10, 25, 50, 100, 250, 500, 1000, 2500, 5000, 10000]

 def __init__(self, name: str):
 self.name = name
 self._counts = [0] * (len(self.BUCKETS) + 1) # +1 for overflow
 self._sum = 0.0
 self._count = 0

 def observe(self, value_ms: float) ->None:
 self._sum += value_ms
 self._count += 1
 for i, threshold in enumerate(self.BUCKETS):
 if value_ms <= threshold:
 self._counts[i] += 1
 return
 self._counts[-1] += 1 # Overflow bucket

 @property
 def avg(self) ->float:
 return self._sum / self._count if self._count else 0

 @property
 def p50(self) ->float:
 return self._percentile(0.5)

 @property
 def p99(self) ->float:
 return self._percentile(0.99)

 def _percentile(self, p: float) ->float:
 target = int(self._count * p)
 cumulative = 0
 for i, bucket_count in enumerate(self._counts):
 cumulative += bucket_count
 if cumulative >= target:
 if i < len(self.BUCKETS):
 return float(self.BUCKETS[i])
 return float("inf")
 return float("inf")

model_latency = Histogram("model_call_duration_ms")
for ms in [340, 620, 780, 840, 1200, 2100, 450, 920, 670, 510]:
 model_latency.observe(ms)

print(f"avg={model_latency.avg:.0f}ms p50={model_latency.p50}ms p99={model_latency.p99}ms")
# avg=843ms p50=500ms p99=2500ms

 def inc(self, n: float = 1) ->None:

4. Summary — Multi-Percentile Snapshots

>
class Summary:
 """For metrics where you need p50/p90/p99 at query time."""

 def __init__(self, name: str, max_samples: int = 1000):
 self.name = name
 self._samples: list[float] = []
 self._max = max_samples

 def observe(self, value: float) ->None:
 if len(self._samples) >= self._max:
 # Reservoir sampling — replace random element
 idx = int(time.time() * 1000) % self._max
 self._samples[idx] = value
 else:
 self._samples.append(value)

 def percentiles(self, ps: list[float] = [0.5, 0.9, 0.99]) ->dict[str, float]:
 sorted_samples = sorted(self._samples)
 result = {}
 for p in ps:
 idx = int(len(sorted_samples) * p)
 idx = min(idx, len(sorted_samples) - 1)
 result[f"p{int(p*100)}"] = sorted_samples[idx]
 return result

token_usage = Summary("tokens_per_turn")
for tokens in [120, 340, 890, 1200, 2100, 450, 670, 920, 1500, 2100]:
 token_usage.observe(tokens)

print(token_usage.percentiles())
# {'p50': 890, 'p90': 2100, 'p99': 2100}

Metrics Registry Pattern

>
class MetricsRegistry:
 """Singleton registry — collect all metrics in one place."""

 def __init__(self):
 self._counters: dict[str, Counter] = {}
 self._histograms: dict[str, Histogram] = {}
 self._gauges: dict[str, Gauge] = {}
 self._summaries: dict[str, Summary] = {}

 def counter(self, name: str) ->Counter:
 if name not in self._counters:
 self._counters[name] = Counter(name)
 return self._counters[name]

 def histogram(self, name: str) ->Histogram:
 if name not in self._histograms:
 self._histograms[name] = Histogram(name)
 return self._histograms[name]

 def gauge(self, name: str) ->Gauge:
 if name not in self._gauges:
 self._gauges[name] = Gauge(name)
 return self._gauges[name]

 def summary(self, name: str) ->Summary:
 if name not in self._summaries:
 self._summaries[name] = Summary(name)
 return self._summaries[name]

 def export(self) ->dict:
 """Export snapshot for dashboards."""
 return {
 "counters": {n: {"value": c.value, "by_label": c.by_label()}
 for n, c in self._counters.items()},
 "histograms": {n: {"avg": h.avg, "p50": h.p50, "p99": h.p99,
 "count": h._count}
 for n, h in self._histograms.items()},
 "gauges": {n: g.value for n, g in self._gauges.items()},
 "summaries": {n: s.percentiles() for n, s in self._summaries.items()},
 }

registry = MetricsRegistry()
>
Memory cost:Histograms use ~100 bytes each (11 ints + sum + count). 100 tracked metrics = ~10KB. Summaries with 1000 samples = 8KB each. Total for a typical agent: under 100KB.

Layer 3: Structured Logging (The "Why" Layer)

>

Logs connect traces and metrics to human-readable context. The key insight: log as structured JSON, not free text.

>
import logging
import json
import sys

class StructuredFormatter(logging.Formatter):
 """JSON Lines output — one log event per line."""

 def format(self, record: logging.LogRecord) ->str:
 log_entry = {
 "timestamp": self.formatTime(record),
 "level": record.levelname,
 "message": record.getMessage(),
 "module": record.module,
 "line": record.lineno,
 }

 # Add trace context if available
 span = _current_span.get(None)
 if span:
 log_entry["span_id"] = span.span_id
 log_entry["span_name"] = span.name

 # Add custom attributes
 if hasattr(record, "agent_data"):
 log_entry["data"] = record.agent_data

 return json.dumps(log_entry)


# Setup
handler = logging.StreamHandler(sys.stdout)
handler.setFormatter(StructuredFormatter())

logger = logging.getLogger("agent")
logger.setLevel(logging.INFO)
logger.addHandler(handler)

# Usage in agent loop
class _AgentLogAdapter(logging.LoggerAdapter):
 def process(self, msg, kwargs):
 # Attach extra data to every log
 extra = kwargs.setdefault("extra", {})
 extra.setdefault("agent_data", {})

 # Inject trace context
 span = _current_span.get(None)
 if span:
 extra["agent_data"]["span_id"] = span.span_id

 return msg, kwargs

agent_logger = _AgentLogAdapter(logger, {})

# Now logs look like:
# {"timestamp":"2026-07-10T...","level":"INFO","message":"Tool call resolved",
# "span_id":"a3f4b2c1","data":{"tool":"read_file","duration_ms":4,"success":true}}
agent_logger.info("Tool call resolved", extra={
 "agent_data": {"tool": "read_file", "duration_ms": 4, "success": True}
})

Wiring It All Together: The Agent Instrumented Turn

>
def instrumented_agent_turn(
 user_message: str,
 model: str,
 tracer: AgentTracer,
 metrics: MetricsRegistry,
) ->str:
 """One agent turn with full observability instrumentation."""

 turn_span = tracer.start_span(
 "agent_turn",
 model=model,
 user_msg_len=len(user_message),
 )

 metrics.gauge("agent_concurrent_turns").inc()
 metrics.counter("agent_turns_total").inc(model=model)

 try:
 # --- Model call ---
 with tracer_span("model_call", model=model):
 model_start = time.monotonic()

 response = call_model(user_message, model=model)

 model_ms = (time.monotonic() - model_start) * 1000
 metrics.histogram("model_call_duration_ms").observe(model_ms)
 metrics.summary("tokens_input").observe(response.input_tokens)
 metrics.summary("tokens_output").observe(response.output_tokens)
 metrics.counter("model_tokens_total").inc(
 n=response.input_tokens + response.output_tokens,
 direction="input",
 model=model,
 )
 metrics.counter("model_tokens_total").inc(
 n=response.output_tokens,
 direction="output",
 model=model,
 )

 if response.cache_hit:
 metrics.counter("model_cache_hits").inc()

 # --- Tool calls ---
 if response.tool_calls:
 metrics.gauge("agent_concurrent_tool_calls").inc(
 len(response.tool_calls)
 )

 for tc in response.tool_calls:
 tool_span = tracer.start_span(
 "tool_call",
 tool_name=tc.name,
 args_size=len(str(tc.args)),
 )

 tool_start = time.monotonic()
 try:
 result = execute_tool(tc.name, tc.args)
 tool_ms = (time.monotonic() - tool_start) * 1000

 metrics.histogram("tool_duration_ms",
 tool=tc.name).observe(tool_ms)
 metrics.counter("tool_calls_total").inc(
 tool=tc.name, status="success"
 )

 tool_span.set_attribute("result_size", len(str(result)))
 tool_span.set_attribute("success", True)

 except Exception as e:
 tool_ms = (time.monotonic() - tool_start) * 1000
 metrics.histogram("tool_duration_ms",
 tool=tc.name).observe(tool_ms)
 metrics.counter("tool_calls_total").inc(
 tool=tc.name, status="error", error=type(e).__name__
 )
 metrics.counter("tool_errors_total").inc(
 tool=tc.name, error=type(e).__name__
 )

 tool_span.add_event("error",
 type=type(e).__name__,
 message=str(e)[:200])
 tool_span.set_attribute("success", False)

 agent_logger.error("Tool call failed",
 extra={"agent_data": {
 "tool": tc.name,
 "error": str(e)[:200],
 "duration_ms": tool_ms,
 }})

 finally:
 tracer.end_span(tool_span)

 metrics.gauge("agent_concurrent_tool_calls").dec(
 len(response.tool_calls)
 )

 turn_ms = (time.monotonic() - turn_span.start_time) * 1000
 metrics.histogram("turn_duration_ms").observe(turn_ms)

 return response.content

 finally:
 tracer.end_span(turn_span)
 metrics.gauge("agent_concurrent_turns").dec()

Common Pitfalls

>
Pitfall 1: Sampling too aggressively.If you sample 1% of turns, you miss the rare failure modes that only appear in long conversations. Sample 100% of turns, but limit span attributes to 500 bytes each.
>
Pitfall 2: Blocking I/O in trace export.Writing spans to disk/network synchronously adds latency to every tool call. Buffer in memory, flush every 30 seconds in a background thread.
>
Pitfall 3: Too many label cardinalities.A counter with user_idas a label creates one time series per user. With 10k users, your registry balloons to 10MB. Limit labels to low-cardinality values (model name, tool name, error type).
>

Cost of Full Instrumentation

>
MetricOverhead per turnMemory per 1k turns
Span creation (avg 15 spans/turn)~45µs~15KB
Metrics observation (avg 40 ops/turn)~120µs~8KB
Structured logging (5 log lines/turn)~200µs~5KB on disk
Total~0.4ms~28KB
>

For a turn that takes 1-5 seconds of model latency, 0.4ms of observability overhead is 0.01-0.04%. There is no reason not to instrument.

Conclusion

>

Agent observability reduces to three questions:

>
  1. What happened?→ Distributed traces (span trees)
  2. How bad?→ Metrics (counters, histograms, gauges)
  3. Why?→ Structured logs (JSON with trace context)
>

The implementation is simpler than you think — Python's contextvars, dataclasses, and jsonare all you need. The entire system above is under 400 lines of code, adds 0.4ms per turn, and lets you debug agent behavior without guessing.

>

The next time your agent loops on a tool failure, you won't be reading console logs at 2am — you'll be looking at a span tree that shows exactly which call timed out, how many retries happened, and what the model was thinking at each step.
