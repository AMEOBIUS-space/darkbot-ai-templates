Building a Tool Registry for AI Agents: Dynamic Discovery, Routing, and Usage Analytics

>

AI AgentsTool RegistryPythonMCP

>

An agent with 50 tools is like a library with no card catalog — the books exist, but nobody can find them. The model wastes context window space reading schemas for tools it'll never call. It picks the wrong tool for the job. It hallucinates parameters for tools that almost match.

>

A tool registry solves this. It's a dynamic layer between "all tools that exist" and "tools available for this specific turn." This guide walks through building one in pure Python — zero dependencies, 300 lines of code.

>

What a Tool Registry Does

>
ProblemWithout RegistryWith Registry
Too many tools in context50 schemas × 500 bytes = 25KB wasted per turnOnly relevant 5-8 tools loaded
Wrong tool selectedModel picks search_webwhen it needs search_filesRegistry suggests based on query intent
Dead toolsAPI key expired, tool silently failsHealth checks catch it, registry disables
No usage analytics"Which tools are actually used?" — no ideaPer-tool call counts, success rates, latency
Version conflictsTwo tools with same name, different schemasNamespacing + priority resolution

Step 1: Tool Registration

>

Each tool registers itself with a name, schema, handler, and metadata:

>
from dataclasses import dataclass, field
from typing import Any, Callable
import time
import json

@dataclass
class ToolEntry:
 name: str
 description: str
 handler: Callable
 schema: dict[str, Any]
 namespace: str = "core"
 priority: int = 0 # Higher = preferred on conflict
 requires_env: list[str] = field(default_factory=list)
 requires_path: list[str] = field(default_factory=list)
 tags: list[str] = field(default_factory=list)
 version: str = "1.0.0"

 # Runtime metadata (populated by registry)
 call_count: int = 0
 error_count: int = 0
 last_called: float | None = None
 avg_latency_ms: float = 0.0
 enabled: bool = True
 health_status: str = "unknown" # unknown | healthy | degraded | down


class ToolRegistry:
 def __init__(self):
 self._tools: dict[str, ToolEntry] = {}
 self._aliases: dict[str, str] = {} # alias → canonical name
 self._latency_samples: dict[str, list[float]] = {}

 def register(self, entry: ToolEntry, aliases: list[str] | None = None) ->None:
 # Conflict resolution by priority
 if entry.name in self._tools:
 existing = self._tools[entry.name]
 if entry.priority <= existing.priority:
 # Keep existing, warn
 print(f"Warning: {entry.name} already registered "
 f"with higher priority ({existing.priority})")
 return

 self._tools[entry.name] = entry

 if aliases:
 for alias in aliases:
 self._aliases[alias] = entry.name

 def get(self, name: str) ->ToolEntry | None:
 # Resolve alias
 canonical = self._aliases.get(name, name)
 tool = self._tools.get(canonical)
 if tool and tool.enabled and tool.health_status != "down":
 return tool
 return None

 def list_tools(self, namespace: str | None = None,
 tags: list[str] | None = None,
 only_enabled: bool = True) ->list[ToolEntry]:
 tools = list(self._tools.values())

 if namespace:
 tools = [t for t in tools if t.namespace == namespace]
 if tags:
 tag_set = set(tags)
 tools = [t for t in tools if tag_set & set(t.tags)]
 if only_enabled:
 tools = [t for t in tools if t.enabled]

 return tools

Step 2: Usage Tracking

>

Every call updates the tool's runtime metadata. This data feeds back into routing decisions:

>
class ToolRegistry:
 # ... continued ...

 def call(self, name: str, args: dict, task_id: str | None = None) ->str:
 tool = self.get(name)
 if not tool:
 return json.dumps({"error": f"Tool '{name}' not found or disabled"})

 start = time.monotonic()
 try:
 result = tool.handler(args, task_id=task_id)
 elapsed_ms = (time.monotonic() - start) * 1000

 # Update metadata
 tool.call_count += 1
 tool.last_called = time.time()
 self._update_latency(name, elapsed_ms)

 return result

 except Exception as e:
 elapsed_ms = (time.monotonic() - start) * 1000
 tool.error_count += 1
 tool.call_count += 1
 self._update_latency(name, elapsed_ms)

 return json.dumps({
 "error": str(e),
 "tool": name,
 "type": type(e).__name__,
 })

 def _update_latency(self, name: str, latency_ms: float) ->None:
 if name not in self._latency_samples:
 self._latency_samples[name] = []

 samples = self._latency_samples[name]
 samples.append(latency_ms)

 # Keep last 100 samples (rolling window)
 if len(samples) >100:
 samples.pop(0)

 # Update average incrementally
 tool = self._tools[name]
 tool.avg_latency_ms = sum(samples) / len(samples)

 def get_stats(self) ->dict[str, dict]:
 """Per-tool usage statistics."""
 stats = {}
 for name, tool in self._tools.items():
 samples = self._latency_samples.get(name, [])
 p99 = sorted(samples)[int(len(samples) * 0.99)] if samples else 0

 stats[name] = {
 "calls": tool.call_count,
 "errors": tool.error_count,
 "error_rate": tool.error_count / max(tool.call_count, 1),
 "avg_latency_ms": round(tool.avg_latency_ms, 2),
 "p99_latency_ms": round(p99, 2),
 "last_called": tool.last_called,
 "health": tool.health_status,
 }

 return stats

Step 3: Health Checks

>

Tools depend on external resources — API keys, file paths, network access. The registry periodically checks if tools are still functional:

>
import os

class ToolRegistry:
 # ... continued ...

 def check_health(self) ->dict[str, str]:
 """Check all tools' prerequisites. Update health_status."""

 results = {}
 for name, tool in self._tools.items():
 if not tool.enabled:
 results[name] = "disabled"
 continue

 # Check environment variables
 missing_env = [
 env for env in tool.requires_env
 if not os.getenv(env)
 ]

 # Check required paths
 missing_paths = [
 p for p in tool.requires_path
 if not os.path.exists(p)
 ]

 if missing_env or missing_paths:
 tool.health_status = "down"
 results[name] = f"down (missing: {', '.join(missing_env + missing_paths)})"
 else:
 # Optionally run a custom health check
 tool.health_status = "healthy"
 results[name] = "healthy"

 return results

 def disable(self, name: str) ->None:
 """Disable a tool — removes it from routing."""
 if name in self._tools:
 self._tools[name].enabled = False

 def enable(self, name: str) ->None:
 """Re-enable a previously disabled tool."""
 if name in self._tools:
 self._tools[name].enabled = True

Step 4: Intelligent Routing

>

The router decides which tools to present to the model based on the current query. Three strategies, from simple to advanced:

>

Strategy 1: Tag-Based Filtering

>
class ToolRouter:
 """Select tools based on tags and query context."""

 def __init__(self, registry: ToolRegistry):
 self.registry = registry

 def select_by_context(self, context_type: str) ->list[ToolEntry]:
 """Select tools relevant to a context type."""

 tag_map = {
 "coding": ["file", "code", "terminal", "search"],
 "web": ["browser", "http", "scrape", "search"],
 "data": ["sql", "csv", "json", "chart"],
 "communication": ["email", "message", "notify"],
 "system": ["process", "file", "env", "config"],
 }

 tags = tag_map.get(context_type, [])
 return self.registry.list_tools(tags=tags, only_enabled=True)
>

Strategy 2: Keyword Matching

>
class ToolRouter:
 # ... continued ...

 def select_by_query(self, query: str, max_tools: int = 10) ->list[ToolEntry]:
 """Select tools whose name/description/tags match the query."""

 query_lower = query.lower()
 scored: list[tuple[int, ToolEntry]] = []

 for tool in self.registry.list_tools(only_enabled=True):
 score = 0

 # Exact name match: highest score
 if tool.name.lower() in query_lower:
 score += 100

 # Description keywords
 desc_words = set(tool.description.lower().split())
 query_words = set(query_lower.split())
 overlap = desc_words & query_words
 score += len(overlap) * 10

 # Tag matches
 for tag in tool.tags:
 if tag.lower() in query_lower:
 score += 15

 # Penalize tools with high error rates
 stats = self.registry.get_stats().get(tool.name, {})
 error_rate = stats.get("error_rate", 0)
 score -= int(error_rate * 50)

 # Boost frequently-used tools (popularity signal)
 call_count = stats.get("calls", 0)
 if call_count >0:
 score += min(call_count // 10, 20)

 if score >0:
 scored.append((score, tool))

 # Sort by score descending, take top N
 scored.sort(key=lambda x: x[0], reverse=True)
 return [tool for _, tool in scored[:max_tools]]
>

Strategy 3: Schema-Based (Most Precise)

>
class ToolRouter:
 # ... continued ...

 def select_by_schema_fit(self, query: str,
 available_args: dict,
 max_tools: int = 8) ->list[ToolEntry]:
 """Select tools whose parameter schemas match available arguments."""

 scored: list[tuple[int, ToolEntry]] = []

 for tool in self.registry.list_tools(only_enabled=True):
 params = tool.schema.get("parameters", {})
 required = set(params.get("required", []))
 properties = set(params.get("properties", {}).keys())

 # How many required params do we have args for?
 have_args = set(available_args.keys())
 required_match = len(required & have_args)
 optional_match = len(properties & have_args) - required_match

 # Score: required param matches are worth more
 score = required_match * 30 + optional_match * 10

 # Also do keyword matching
 query_lower = query.lower()
 if tool.name.lower() in query_lower:
 score += 50

 if score >0:
 scored.append((score, tool))

 scored.sort(key=lambda x: x[0], reverse=True)
 return [tool for _, tool in scored[:max_tools]]

Step 5: Dynamic Toolset Construction

>

Instead of sending all 50 tools to the model every turn, construct a per-turn toolset:

>
class AgentToolManager:
 """Manages tool selection per turn — keeps context window lean."""

 def __init__(self, registry: ToolRegistry, router: ToolRouter):
 self.registry = registry
 self.router = router
 self._always_include: set[str] = set()
 self._always_exclude: set[str] = set()

 def always_include(self, name: str) ->None:
 self._always_include.add(name)

 def always_exclude(self, name: str) ->None:
 self._always_exclude.add(name)

 def build_toolset(self, query: str,
 context_type: str | None = None,
 max_tools: int = 15) ->list[dict]:
 """Build the tool schema list for this turn's model call."""

 # Start with always-included tools
 selected: dict[str, ToolEntry] = {}

 for name in self._always_include:
 tool = self.registry.get(name)
 if tool:
 selected[name] = tool

 # Add context-based tools
 if context_type:
 for tool in self.router.select_by_context(context_type):
 if tool.name not in self._always_exclude:
 selected[tool.name] = tool

 # Add query-matched tools
 for tool in self.router.select_by_query(query, max_tools=max_tools):
 if tool.name not in self._always_exclude:
 selected[tool.name] = tool

 # Convert to OpenAI tool schema format
 tools = []
 for tool in list(selected.values())[:max_tools]:
 tools.append({
 "type": "function",
 "function": {
 "name": tool.name,
 "description": tool.description,
 "parameters": tool.schema,
 }
 })

 return tools

 def call_tool(self, name: str, args: dict,
 task_id: str | None = None) ->str:
 """Execute a tool call and track metrics."""
 return self.registry.call(name, args, task_id)

Context Window Savings

>
A typical tool schema is 200-600 bytes. With 50 tools, that's 10-30KB per turn. With intelligent routing, you load 8-15 relevant tools — saving 60-80% of schema context.
>
ScenarioTools in ContextSchema Bytes% Saved
All tools (no registry)50~20,000—
Tag-based filter15~6,00070%
Query + tag routing8~3,20084%
Schema-fit routing5~2,00090%
>

At GPT-4o pricing ($2.50/1M input tokens), saving 18KB per turn across 1,000 turns/day saves ~$90/month — just from smarter tool selection.

>

Common Pitfalls

>
Pitfall 1: Over-filtering.If the router is too aggressive, the model can't access tools it genuinely needs. Always include a fallback set of core tools (read_file, terminal, search).
>
Pitfall 2: Stale health checks.A health check run at startup doesn't reflect runtime state. Re-check every N turns, especially for tools depending on external APIs.
>
Pitfall 3: Breaking prompt caching.If the toolset changes every turn, the model's prompt cache misses. Use stable toolsets within a conversation — re-route only when context type changes.

Conclusion

>

A tool registry is the difference between an agent that flails and an agent that acts with purpose. The implementation is simple — three classes (Registry, Router, Manager), 300 lines, zero dependencies.

>

The key insight: tool selection is a first-class problem. Dumping all tools into the system prompt and hoping the model picks right is like throwing a toolbox at someone and yelling "figure it out." A registry curates the right tools for each moment, tracks what works, and keeps the context window lean.
