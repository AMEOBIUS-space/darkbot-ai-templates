Why Agent Testing Is Different from Unit Testing

Unit tests assert exact outputs. Agent outputs are non-deterministic — the same prompt can produce different valid responses. You need behavioral testing: does the output satisfy the task constraints, not does it match a string?

>
Traditional Unit TestAgent Evaluation Test
assert add(2, 3) == 5assert response.contains_code_block() and response.uses_type_hints()
Exact matchConstraint satisfaction
Binary pass/failMulti-dimensional score (0.0-1.0)
DeterministicProbabilistic (run N times, average)
Fast (microseconds)Slow (API calls, 1-5s each)

Step 1: Test Case Definition

from dataclasses import dataclass, field
from typing import Callable, Any

@dataclass
class TestCase:
 test_id: str
 category: str # "coding", "reasoning", "safety", "tools"
 prompt: str
 constraints: list[Callable[[str], tuple[bool, str]]] = field(default_factory=list)
 min_pass_rate: float = 0.8 # Fraction of constraints that must pass
 metadata: dict[str, Any] = field(default_factory=dict)


# Constraint builders — reusable assertions
def contains_code() ->Callable:
 def check(response: str) ->tuple[bool, str]:
 has_code = "```" in response or " " in response
 return (has_code, "Response must contain code block" if not has_code else "")
 return check

def uses_type_hints() ->Callable:
 def check(response: str) ->tuple[bool, str]:
 indicators = ["->", ": str", ": int", ": float", ": bool", ": list",
 ": dict", " | ", "Optional[", "Union["]
 found = sum(1 for i in indicators if i in response)
 return (found >= 2, f"Need 2+ type hint indicators, found {found}")
 return check

def max_length(max_chars: int) ->Callable:
 def check(response: str) ->tuple[bool, str]:
 return (len(response) <= max_chars,
 f"Response {len(response)} > {max_chars} chars")
 return check

def contains_all(terms: list[str]) ->Callable:
 def check(response: str) ->tuple[bool, str]:
 lower = response.lower()
 missing = [t for t in terms if t.lower() not in lower]
 return (not missing, f"Missing terms: {missing}" if missing else "")
 return check

def excludes(terms: list[str]) ->Callable:
 def check(response: str) ->tuple[bool, str]:
 lower = response.lower()
 found = [t for t in terms if t.lower() in lower]
 return (not found, f"Found forbidden terms: {found}" if found else "")
 return check

def follows_format(format_type: str) ->Callable:
 def check(response: str) ->tuple[bool, str]:
 if format_type == "json":
 import json
 try:
 json.loads(response)
 return (True, "")
 except Exception:
 return (False, "Response is not valid JSON")
 elif format_type == "markdown":
 has_header = response.startswith("#")
 has_code = "```" in response
 return (has_header or has_code,
 "Response should have markdown structure")
 return (True, "")
 return check

Step 2: The Test Runner

import time
import json
from dataclasses import asdict

@dataclass
class TestResult:
 test_id: str
 category: str
 passed: bool
 score: float # 0.0-1.0 fraction of constraints passed
 constraint_results: list[dict]
 response: str
 latency_ms: float
 tokens_used: int = 0
 timestamp: float = field(default_factory=time.time)


class EvalHarness:
 def __init__(self):
 self._tests: list[TestCase] = []
 self._results: list[TestResult] = []
 self._regression_baseline: dict[str, float] = {}

 def add_test(self, test: TestCase) ->None:
 self._tests.append(test)

 def run_test(self, test: TestCase,
 agent_fn: Callable[[str], str],
 model: str = "") ->TestResult:
 """Run a single test case against an agent."""
 start = time.monotonic()
 response = agent_fn(test.prompt)
 latency_ms = (time.monotonic() - start) * 1000

 # Evaluate constraints
 constraint_results = []
 passed_count = 0

 for constraint in test.constraints:
 ok, message = constraint(response)
 constraint_results.append({
 "passed": ok,
 "message": message,
 })
 if ok:
 passed_count += 1

 total = len(test.constraints)
 score = passed_count / total if total >0 else 1.0
 passed = score >= test.min_pass_rate

 result = TestResult(
 test_id=test.test_id,
 category=test.category,
 passed=passed,
 score=score,
 constraint_results=constraint_results,
 response=response[:500], # Truncate for storage
 latency_ms=latency_ms,
 )
 self._results.append(result)
 return result

 def run_suite(self, agent_fn: Callable[[str], str],
 categories: list[str] | None = None,
 repeats: int = 1) ->dict:
 """Run all tests (optionally filtered by category)."""
 tests = self._tests
 if categories:
 tests = [t for t in tests if t.category in categories]

 for _ in range(repeats):
 for test in tests:
 self.run_test(test, agent_fn)

 return self.summarize()

 def summarize(self) ->dict:
 """Aggregate results by category."""
 by_category: dict[str, list[TestResult]] = {}
 for r in self._results:
 by_category.setdefault(r.category, []).append(r)

 summary = {}
 for cat, results in by_category.items():
 scores = [r.score for r in results]
 summary[cat] = {
 "test_count": len(results),
 "pass_rate": sum(1 for r in results if r.passed) / len(results),
 "avg_score": sum(scores) / len(scores),
 "min_score": min(scores),
 "max_score": max(scores),
 "avg_latency_ms": sum(r.latency_ms for r in results) / len(results),
 }

 overall_pass = sum(1 for r in self._results if r.passed)
 summary["_overall"] = {
 "total_tests": len(self._results),
 "overall_pass_rate": overall_pass / max(len(self._results), 1),
 "categories": list(by_category.keys()),
 }

 return summary

Step 3: Regression Detection

class RegressionDetector:
 """Compare current results against a baseline to detect regressions."""

 def __init__(self, harness: EvalHarness):
 self.harness = harness

 def set_baseline(self) ->dict:
 """Snapshot current scores as the regression baseline."""
 summary = self.harness.summarize()
 baseline = {}
 for cat, stats in summary.items():
 if cat != "_overall":
 baseline[cat] = stats["avg_score"]
 self.harness._regression_baseline = baseline
 return baseline

 def check_regression(self, threshold: float = 0.05) ->dict:
 """Compare current results to baseline. Flag regressions >threshold."""
 summary = self.harness.summarize()
 baseline = self.harness._regression_baseline

 if not baseline:
 return {"status": "no_baseline", "message": "Set a baseline first"}

 regressions = []
 improvements = []

 for cat, current_score in summary.items():
 if cat == "_overall" or cat not in baseline:
 continue

 base_score = baseline[cat]
 delta = current_score["avg_score"] - base_score

 if delta < -threshold:
 regressions.append({
 "category": cat,
 "baseline": round(base_score, 3),
 "current": round(current_score["avg_score"], 3),
 "delta": round(delta, 3),
 "severity": "critical" if delta < -0.10 else "warning",
 })
 elif delta >threshold:
 improvements.append({
 "category": cat,
 "baseline": round(base_score, 3),
 "current": round(current_score["avg_score"], 3),
 "delta": round(delta, 3),
 })

 return {
 "status": "regression" if regressions else "ok",
 "regressions": regressions,
 "improvements": improvements,
 "unaffected": [
 cat for cat in baseline
 if cat not in {r["category"] for r in regressions}
 and cat not in {i["category"] for i in improvements}
 ],
 }

Multi-Dimensional Scoring

Real agents need scoring beyond pass/fail. Five dimensions to track:

DimensionWhat It MeasuresConstraint Example
CorrectnessDoes the output solve the problem?Contains expected keywords/logic
CompletenessDoes it address all parts of the prompt?Multiple constraint checks
ConcisenessIs it appropriately concise?Max length constraint
SafetyDoes it avoid harmful content?Excludes dangerous patterns
FormatDoes it follow output format?JSON/markdown/code block check
>

Test Pyramid for Agents

>┌───────────┐
 │ E2E (5) │ ← Full agent sessions (slow, expensive)
 └─────┬─────┘
 ┌───────┴───────┐
 │ Integration(20)│ ← Multi-turn conversations
 └───────┬───────┘
 ┌─────────┴─────────┐
 │ Constraint (50) │ ← Single-prompt constraint checks
 └───────────────────┘
>

Run constraint tests on every prompt change. Run integration tests on every model change. Run E2E tests before release.

>

Common Pitfalls

Pitfall 1: Testing the same prompt you used to build.If you craft the prompt until it works on your test cases, you're testing memorization, not generalization. Use held-out test cases the author hasn't seen.
Pitfall 2: Ignoring latency in the score.An agent that takes 30 seconds per response might be accurate but unusable. Include latency thresholds as a test dimension.
Pitfall 3: Too few repeats.Non-deterministic outputs mean a single run is noise. Run each test 3-5 times and average. A test that passes 4/5 times is more informative than one that passes 1/1.

Conclusion

An evaluation harness is how you know your agent changes are improvements, not regressions. The system above — test cases, constraint builders, multi-dimensional scoring, regression detection — is 300 lines of pure Python.

The discipline matters more than the code: always set a baseline before changing anything. Without a baseline, you can't detect regressions — you're just guessing.
