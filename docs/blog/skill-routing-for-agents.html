Skill Routing: How AI Agents Choose the Right Tool

>

MCPArchitecturePython

>

2026-07-09

>

When an agent has 50 tools available, every API call ships 50 tool schemas in the system prompt. That is 4,000 tokens of overhead before the agent even reads the user's message. Multiply by 30 turns in a conversation and you have burned 120,000 tokens on schema alone.

>

Skill routing fixes this by loading only the tools relevant to the current task. A web scraping question loads browser tools. A git question loads version control tools. Everything else stays unloaded.

>

How Routing Works

>
from dataclasses import dataclass, field
import re

@dataclass
class Skill:
 name: str
 tools: list # tool names to load
 keywords: list # trigger keywords
 patterns: list # regex patterns for task matching
 priority: int = 0 # higher = preferred on conflict

@dataclass
class SkillRouter:
 skills: list = field(default_factory=list)
 
 def register(self, skill: Skill):
 self.skills.append(skill)
 self.skills.sort(key=lambda s: s.priority, reverse=True)
 
 def route(self, task: str) ->dict:
 """Match a task description to the best skill set."""
 task_lower = task.lower()
 scores = []
 
 for skill in self.skills:
 score = 0
 # Keyword matching
 for kw in skill.keywords:
 if kw.lower() in task_lower:
 score += 2
 # Pattern matching 
 for pat in skill.patterns:
 if re.search(pat, task, re.IGNORECASE):
 score += 5
 # Priority bonus
 score += skill.priority
 if score >0:
 scores.append((skill, score))
 
 if not scores:
 return {"skill": "default", "tools": [], "reason": "no match"}
 
 scores.sort(key=lambda x: x[1], reverse=True)
 best = scores[0][0]
 
 return {
 "skill": best.name,
 "tools": best.tools,
 "score": scores[0][1],
 "alternatives": [(s.name, sc) for s, sc in scores[1:3]]
 }

Defining Skill Sets

>
router = SkillRouter()

# Web scraping skill
router.register(Skill(
 name="web_scraping",
 tools=["browser_navigate", "browser_click", "browser_snapshot",
 "browser_type", "web_extract", "terminal"],
 keywords=["scrape", "crawl", "extract", "parse", "website",
 "html", "page", "url", "browser"],
 patterns=[r"https?://\S+", r"scrap[ei]\b\w*", r"crawl\w*"],
 priority=10
))

# Git operations skill
router.register(Skill(
 name="git_ops",

 "pull request", "pr", "diff"],

# File editing skill

 'score': 12,
 'alternatives': [('file_edit', 5)]}
>
The router returns alternatives so the agent can request a skill switch mid-task. If a testing task reveals a code bug, the agent can load file_edittools without a full re-route.

Multi-Skill Routing

>

Some tasks span multiple skills. A task like "scrape data from a website and save to SQLite" needs both web scraping and database tools:

>
def route_multi(self, task: str, max_skills: int = 3) ->dict:
 """Route to multiple skills, merge tool sets."""
 results = []
 for skill in self.skills:
 score = self._score(skill, task)
 if score >0:
 results.append((skill, score))
 
 results.sort(key=lambda x: x[1], reverse=True)
 
 # Merge tools from top N skills
 merged_tools = set()
 selected = []
 for skill, score in results[:max_skills]:
 merged_tools.update(skill.tools)
 selected.append({"name": skill.name, "score": score})
 
 return {
 "skills": selected,
 "tools": list(merged_tools),
 "total_tools": len(merged_tools)
 }

Fallback Strategy

>

When no skill matches, load a minimal safe set:

>
DEFAULT_TOOLS = ["terminal", "read_file", "search_files", "web_search"]

# On no match:
result = {"skill": "default", "tools": DEFAULT_TOOLS}
# The agent can still operate — just with basic tools

Dynamic Skill Discovery

>

For plugin systems, discover skills at runtime from a directory:

>
import importlib
from pathlib import Path

def discover_skills(plugin_dir: str) ->list:
 """Load skill definitions from Python files."""
 skills = []
 for pyfile in Path(plugin_dir).glob("skill_*.py"):
 module = importlib.import_module(pyfile.stem)
 if hasattr(module, "SKILL"):
 skills.append(module.SKILL)
 return skills

# Each skill_*.py file defines:
# SKILL = Skill(name="...", tools=[...], keywords=[...], patterns=[...])
