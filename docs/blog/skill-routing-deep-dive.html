What Is a Skill?

A skill is a named collection of tools, prompts, and metadata that handles a category of tasks. Examples:

Step 1: Skill Registry

from dataclasses import dataclass, field
from typing import Any
import re
import json

@dataclass
class Skill:
 name: str
 description: str
 tools: list[str]
 keywords: list[str] # Trigger words in user message
 tags: list[str] # Categorical tags
 system_prompt: str = "" # Additional context for this skill
 priority: int = 0 # Higher = preferred on tie
 metadata: dict[str, Any] = field(default_factory=dict)


class SkillRegistry:
 def __init__(self):
 self._skills: dict[str, Skill] = {}

 def register(self, skill: Skill) ->None:
 self._skills[skill.name] = skill

 def get(self, name: str) ->Skill | None:
 return self._skills.get(name)

 def list_skills(self) ->list[Skill]:
 return list(self._skills.values())

 def all_tools(self) ->set[str]:
 return {t for s in self._skills.values() for t in s.tools}

Step 2: The Router

class SkillRouter:
 """Match a user message to the most relevant skills."""

 def __init__(self, registry: SkillRegistry):
 self.registry = registry

 def route(self, message: str, max_skills: int = 3) ->list[dict]:
 """Return ranked skills with match scores."""
 scored: list[tuple[float, Skill]] = []

 for skill in self.registry.list_skills():
 score = self._score_match(message, skill)
 scored.append((score, skill))

 scored.sort(key=lambda x: (-x[0], -x[1].priority))

 results = []
 for score, skill in scored[:max_skills]:
 if score >0:
 results.append({
 "skill": skill.name,
 "score": round(score, 2),
 "tools": skill.tools,
 "matched": self._what_matched(message, skill),
 })

 return results

 def select_tools(self, message: str,
 max_skills: int = 3,
 always_include: list[str] | None = None) ->list[str]:
 """Select the full tool set for this message."""
 routed = self.route(message, max_skills)

 tools: list[str] = []
 seen: set[str] = set()

 # Always-included tools (core set)
 for t in (always_include or []):
 if t not in seen:
 tools.append(t)
 seen.add(t)

 # Skill-matched tools
 for r in routed:
 for t in r["tools"]:
 if t not in seen:
 tools.append(t)
 seen.add(t)

 return tools

 def _score_match(self, message: str, skill: Skill) ->float:
 """Score how well a skill matches the message (0.0-1.0)."""
 msg_lower = message.lower()
 msg_words = set(msg_lower.split())

 score = 0.0

 # Direct keyword matches (weighted heavily)
 kw_matches = sum(1 for kw in skill.keywords if kw.lower() in msg_lower)
 score += min(kw_matches * 0.15, 0.6)

 # Tag matches (medium weight)
 tag_matches = sum(1 for tag in skill.tags if tag.lower() in msg_lower)
 score += min(tag_matches * 0.10, 0.3)

 # Description word overlap (light weight)
 desc_words = set(skill.description.lower().split())
 overlap = len(msg_words & desc_words)
 score += min(overlap * 0.02, 0.2)

 # Skill name in message (strong signal)
 if skill.name.lower() in msg_lower:
 score += 0.25

 # Apply priority bonus
 score += skill.priority * 0.05

 return min(score, 1.0)

 def _what_matched(self, message: str, skill: Skill) ->dict:
 """Detail what triggered the match (for debugging)."""
 msg_lower = message.lower()

 matched_kw = [kw for kw in skill.keywords if kw.lower() in msg_lower]
 matched_tags = [t for t in skill.tags if t.lower() in msg_lower]
 name_match = skill.name.lower() in msg_lower

 return {
 "keywords": matched_kw,
 "tags": matched_tags,
 "name_match": name_match,
 }

Step 3: Multi-Skill Composition

class MultiSkillComposer:
 """Compose multiple skills for complex tasks."""

 def __init__(self, router: SkillRouter):
 self.router = router

 def compose(self, message: str,
 max_skills: int = 3) ->dict:
 """Build a composed skill set for a complex task."""
 routed = self.router.route(message, max_skills)

 # Merge tools (dedup, preserve order)
 tools: list[str] = []
 seen: set[str] = set()
 for r in routed:
 for t in r["tools"]:
 if t not in seen:
 tools.append(t)
 seen.add(t)

 # Merge system prompts
 prompts = []
 for r in routed:
 skill = self.router.registry.get(r["skill"])
 if skill and skill.system_prompt:
 prompts.append(f"## {skill.name.title()} Mode\n{skill.system_prompt}")

 composed_prompt = "\n\n".join(prompts) if prompts else ""

 # Calculate context savings
 total_possible = len(self.router.registry.all_tools())
 selected = len(tools)
 savings_pct = (1 - selected / max(total_possible, 1)) * 100

 return {
 "skills_active": [r["skill"] for r in routed],
 "tools": tools,
 "tool_count": selected,
 "total_available": total_possible,
 "savings_pct": round(savings_pct, 1),
 "system_prompt": composed_prompt,
 "routing_detail": routed,
 }

Routing with Conversation Context

class ContextualRouter:
 """Router that considers conversation history, not just the latest message."""

 def __init__(self, router: SkillRouter):
 self.router = router
 self._active_skills: set[str] = set()
 self._skill_decay: int = 5 # Turns before skill expires

 def route_with_context(self, message: str,
 history: list[str],
 max_skills: int = 3) ->list[str]:
 """Route considering recent messages too."""
 # Route current message
 current = self.router.route(message, max_skills)

 # Route recent history (lighter weight)
 recent_skills: dict[str, float] = {}
 for i, past_msg in enumerate(history[-3:]):
 weight = 1.0 / (i + 2) # Older = less weight
 for r in self.router.route(past_msg, max_skills):
 recent_skills[r["skill"]] = recent_skills.get(r["skill"], 0) + r["score"] * weight

 # Merge: current message gets full weight, history gets decayed
 all_skills: dict[str, float] = {}
 for r in current:
 all_skills[r["skill"]] = r["score"]

 for skill, score in recent_skills.items():
 all_skills[skill] = all_skills.get(skill, 0) + score * 0.3

 # Select top N
 ranked = sorted(all_skills.items(), key=lambda x: -x[1])
 return [s for s, _ in ranked[:max_skills] if all_skills[s] >0.1]

Common Pitfalls

Pitfall 1: Over-routing.If every message matches every skill, the router adds overhead without savings. Tune keyword lists to be specific. "code" matches too broadly; "fix the bug" is precise.
Pitfall 2: Breaking prompt caching.If the toolset changes every turn, prompt cache misses. Keep routing stable within a conversation — only re-route when the topic clearly changes.
Pitfall 3: Forgetting core tools.Some tools (read_file, terminal) are needed in almost every task. Always include them regardless of routing.

Conclusion

Skill routing is the simplest way to cut agent costs: fewer tools in context means fewer input tokens, better cache hit rates, and more accurate tool selection. The system above — skill registry, scoring router, multi-skill composer — is 250 lines of pure Python with zero dependencies.

The trick is finding the right granularity: too few skills and routing is meaningless; too many and the overhead exceeds the savings. Start with 3-5 broad skills (coding, web, data, messaging, debugging) and split only when routing accuracy suffers.
