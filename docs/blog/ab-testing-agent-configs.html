A/B Testing AI Agent Configurations

>

TestingMCPPython

>

2026-07-09

>

Which model is better for your task: GPT-4o at temperature 0.7 or Claude Sonnet at 0.5? Which prompt produces fewer hallucinated tool calls? Does adding context compression help or hurt accuracy?

>

You cannot answer these questions by trying one config, then trying another, and comparing vibes. You need controlled A/B tests with the same inputs, same evaluation criteria, and statistical significance.

>

The Experiment Framework

>
from dataclasses import dataclass, field
from datetime import datetime
import json, hashlib

@dataclass
class Variant:
 name: str
 config: dict # model, temperature, tools, system_prompt, etc.
 
@dataclass
class TestCase:
 name: str
 input: str
 expected: str = "" # for exact match
 evaluator: str = "manual" # manual, exact, contains, llm_judge
 
@dataclass
class TestResult:
 variant: str
 test_case: str
 output: str
 passed: bool = False
 latency_ms: int = 0
 tokens_used: int = 0
 error: str = ""

@dataclass
class ABTestRunner:
 variants: list = field(default_factory=list)
 test_cases: list = field(default_factory=list)
 results: list = field(default_factory=list)
 
 def add_variant(self, name: str, **config):
 self.variants.append(Variant(name=name, config=config))
 
 def add_test(self, name: str, input: str, expected: str = "", 
 evaluator: str = "contains"):
 self.test_cases.append(
 TestCase(name, input, expected, evaluator)
 )
 
 def evaluate(self, output: str, expected: str, evaluator: str) ->bool:
 if evaluator == "exact":
 return output.strip() == expected.strip()
 elif evaluator == "contains":
 return expected.lower() in output.lower()
 elif evaluator == "manual":
 return True # review later
 return False
 
 def run_pair(self, variant_a: Variant, variant_b: Variant) ->dict:
 """Run all test cases with both variants, compare."""
 results_a = []
 results_b = []
 
 for case in self.test_cases:
 # Run variant A
 out_a = self._run_variant(variant_a, case)
 results_a.append(out_a)
 
 # Run variant B
 out_b = self._run_variant(variant_b, case)
 results_b.append(out_b)
 
 # Compute stats
 pass_a = sum(1 for r in results_a if r.passed)
 pass_b = sum(1 for r in results_b if r.passed)
 tokens_a = sum(r.tokens_used for r in results_a)
 tokens_b = sum(r.tokens_used for r in results_b)
 latency_a = sum(r.latency_ms for r in results_a) // len(results_a)
 latency_b = sum(r.latency_ms for r in results_b) // len(results_b)
 
 return {
 "variant_a": variant_a.name,
 "variant_b": variant_b.name,
 "pass_rate_a": f"{pass_a}/{len(self.test_cases)}",
 "pass_rate_b": f"{pass_b}/{len(self.test_cases)}",
 "tokens_a": tokens_a,
 "tokens_b": tokens_b,
 "avg_latency_a_ms": latency_a,
 "avg_latency_b_ms": latency_b,
 "winner": variant_a.name if pass_a >pass_b else variant_b.name if pass_b >pass_a else "tie"
 }
 
 def _run_variant(self, variant: Variant, case: TestCase) ->TestResult:
 """Execute a single test case with a variant. Override in subclass."""
 # This is where you call your actual agent
 # For demo purposes, return a placeholder
 return TestResult(
 variant=variant.name,
 test_case=case.name,
 output="",
 passed=False
 )

Statistical Significance

>

With 10 test cases, a 9/10 vs 8/10 difference is not significant. You need at least 30 cases for meaningful results:

>
import math

def is_significant(pass_a: int, n_a: int, pass_b: int, n_b: int) ->bool:
 """Quick significance check (normal approximation)."""
 p_a = pass_a / n_a
 p_b = pass_b / n_b
 p_pool = (pass_a + pass_b) / (n_a + n_b)
 
 if p_pool == 0 or p_pool == 1:
 return False
 
 se = math.sqrt(p_pool * (1 - p_pool) * (1/n_a + 1/n_b))
 z = (p_a - p_b) / se
 
 return abs(z) >1.96 # 95% confidence

# Example: 27/30 vs 22/30
print(is_significant(27, 30, 22, 30)) # True — significant
print(is_significant(9, 10, 8, 10)) # False — not enough data
>
Run 30+ test cases per variant. Anything less is directional, not conclusive.

Persistent Results

>
import sqlite3

class PersistentABTest(ABTestRunner):
 def __init__(self, db_path="ab_tests.db"):
 super().__init__()
 self.db = sqlite3.connect(db_path)
 self.db.execute("""
 CREATE TABLE IF NOT EXISTS results (
 experiment TEXT, variant TEXT, test_case TEXT,
 output TEXT, passed INTEGER, tokens INTEGER,
 latency_ms INTEGER, timestamp TEXT
 )
 """)
 self.db.commit()
 
 def save_results(self, experiment_name: str):
 for r in self.results:
 self.db.execute(
 "INSERT INTO results VALUES (?,?,?,?,?,?,?,?)",
 (experiment_name, r.variant, r.test_case, r.output[:500],
 r.passed, r.tokens_used, r.latency_ms, 
 datetime.now().isoformat())
 )
 self.db.commit()
