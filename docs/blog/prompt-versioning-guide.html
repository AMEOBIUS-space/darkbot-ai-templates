Prompt Versioning: Track, Diff, and Roll Back

Prompts MCP Python

2026-07-09

You changed one sentence in the system prompt and suddenly the agent stops calling tools correctly. Was it the tone adjustment? The new instruction order? Without versioning, you are guessing.

Prompt versioning treats system prompts like code: every change is tracked, diffed, and rollback-able.

The Problem

The Version Store

from dataclasses import dataclass, field
from datetime import datetime
import hashlib, sqlite3, json

@dataclass
class PromptVersion:
    version_id: str        # hash of content
    prompt_name: str       # e.g. "system_prompt", "tool_description_search"
    content: str           # the actual prompt text
    author: str = "agent"
    notes: str = ""
    parent_id: str = ""    # previous version hash
    timestamp: str = ""
    active: bool = True
    
    def __post_init__(self):
        if not self.version_id:
            self.version_id = hashlib.sha256(
                f"{self.prompt_name}:{self.content}".encode()
            ).hexdigest()[:12]
        if not self.timestamp:
            self.timestamp = datetime.now().isoformat()

class PromptVersionStore:
    def __init__(self, db_path: str = "prompts.db"):
        self.db = sqlite3.connect(db_path)
        self.db.execute("""
            CREATE TABLE IF NOT EXISTS prompt_versions (
                version_id TEXT PRIMARY KEY,
                prompt_name TEXT,
                content TEXT,
                author TEXT,
                notes TEXT,
                parent_id TEXT,
                timestamp TEXT,
                active INTEGER DEFAULT 1
            )
        """)
        self.db.execute("""
            CREATE INDEX IF NOT EXISTS idx_prompt_name 
            ON prompt_versions(prompt_name)
        """)
        self.db.commit()
    
    def save(self, prompt: PromptVersion) -> str:
        """Save a new version. Returns version_id."""
        # Check if content already exists (dedup)
        existing = self.db.execute(
            "SELECT version_id FROM prompt_versions WHERE version_id = ?",
            (prompt.version_id,)
        ).fetchone()
        
        if existing:
            return existing[0]  # Already saved
        
        # Deactivate previous active version of this prompt
        self.db.execute(
            "UPDATE prompt_versions SET active = 0 WHERE prompt_name = ?",
            (prompt.prompt_name,)
        )
        
        self.db.execute(
            "INSERT INTO prompt_versions VALUES (?,?,?,?,?,?,?,?)",
            (prompt.version_id, prompt.prompt_name, prompt.content,
             prompt.author, prompt.notes, prompt.parent_id,
             prompt.timestamp, 1)
        )
        self.db.commit()
        return prompt.version_id
    
    def get_active(self, prompt_name: str) -> PromptVersion:
        """Get the currently active version of a prompt."""
        row = self.db.execute(
            "SELECT * FROM prompt_versions WHERE prompt_name = ? AND active = 1",
            (prompt_name,)
        ).fetchone()
        if not row:
            return None
        return PromptVersion(
            version_id=row[0], prompt_name=row[1], content=row[2],
            author=row[3], notes=row[4], parent_id=row[5],
            timestamp=row[6], active=bool(row[7])
        )
    
    def get_version(self, version_id: str) -> PromptVersion:
        row = self.db.execute(
            "SELECT * FROM prompt_versions WHERE version_id = ?",
            (version_id,)
        ).fetchone()
        if not row:
            return None
        return PromptVersion(
            version_id=row[0], prompt_name=row[1], content=row[2],
            author=row[3], notes=row[4], parent_id=row[5],
            timestamp=row[6], active=bool(row[7])
        )
    
    def list_versions(self, prompt_name: str) -> list:
        rows = self.db.execute(
            "SELECT version_id, timestamp, author, notes, active "
            "FROM prompt_versions WHERE prompt_name = ? "
            "ORDER BY timestamp DESC",
            (prompt_name,)
        ).fetchall()
        return [{"id": r[0], "time": r[1][:19], "author": r[2], 
                 "notes": r[3], "active": bool(r[4])} for r in rows]
    
    def rollback(self, prompt_name: str, version_id: str) -> bool:
        """Roll back to a specific version."""
        version = self.get_version(version_id)
        if not version:
            return False
        
        # Create a new version with the old content
        new = PromptVersion(
            prompt_name=prompt_name,
            content=version.content,
            notes=f"Rollback to {version_id}",
            parent_id=self.get_active(prompt_name).version_id if self.get_active(prompt_name) else ""
        )
        self.save(new)
        return True

Diffing Prompts

import difflib

def diff_prompts(old: str, new: str) -> str:
    """Generate a readable diff between two prompt versions."""
    old_lines = old.splitlines(keepends=True)
    new_lines = new.splitlines(keepends=True)
    
    diff = difflib.unified_diff(
        old_lines, new_lines,
        fromfile="old", tofile="new",
        n=2  # context lines
    )
    
    return "".join(diff)

Integration with Agent Loop

# At agent startup, load the active prompt
store = PromptVersionStore()
system_prompt = store.get_active("system_prompt").content

# When you update the prompt:
new_prompt = PromptVersion(
    prompt_name="system_prompt",
    content="You are an expert Python developer. Always use tools...",
    notes="Added tool-first instruction, softened tone",
    parent_id=store.get_active("system_prompt").version_id
)
store.save(new_prompt)

# If performance drops, rollback:
store.rollback("system_prompt", "a3f4b2c1d8e9")
Always save the current prompt before editing. If you overwrite without saving, the previous version is lost forever.

Change Impact Tracking

Link prompt versions to agent performance metrics:

def track_impact(store: PromptVersionStore, version_id: str,
                  success_rate: float, avg_cost: float, avg_turns: int):
    """Record performance metrics for a prompt version."""
    store.db.execute("""
        CREATE TABLE IF NOT EXISTS prompt_metrics (
            version_id TEXT, success_rate REAL,
            avg_cost REAL, avg_turns INTEGER,
            timestamp TEXT
        )
    """)
    store.db.execute(
        "INSERT INTO prompt_metrics VALUES (?,?,?,?,?)",
        (version_id, success_rate, avg_cost, avg_turns,
         datetime.now().isoformat())
    )
    store.db.commit()

# After running 30 test cases with prompt v12:
track_impact(store, "v12_hash", success_rate=0.87, 
             avg_cost=0.04, avg_turns=14)

# Compare with v11:
track_impact(store, "v11_hash", success_rate=0.93,
             avg_cost=0.03, avg_turns=11)
