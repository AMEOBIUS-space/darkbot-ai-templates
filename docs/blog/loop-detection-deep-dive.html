Loop Detection for AI Agents: Identify and Break Reasoning Cycles

AI Agents Loop Detection Python MCP

The agent calls search_web("error in auth.py"), reads the results, calls search_web("error in auth.py") again, reads the same results, calls it again... Twenty turns and $2 later, you realize it's stuck in a loop. This guide builds a loop detector that catches this in turn 3, not turn 20.

The Four Loop Types

Loop TypeSignatureExampleFrequency
Exact repeatSame tool + same argssearch_web("error") 3x in a rowCommon
Argument cycleSame tool, args oscillate between 2-3 valuesread_file("a.py") → read_file("b.py") → read_file("a.py")Common
Tool sequence cycleSame sequence of different tools repeatssearch → read → write → search → read → writeModerate
Semantic repeatDifferent tool/args but same intentsearch_web("python error") → search_web("py exception")Rare

Step 1: Action Signatures

import hashlib
import json
from dataclasses import dataclass
from collections import deque

@dataclass
class AgentAction:
    tool_name: str
    args: dict
    result_summary: str = ""
    timestamp: float = 0.0

    def signature(self) -> str:
        """Exact-match signature: tool name + sorted args."""
        args_str = json.dumps(self.args, sort_keys=True)
        return f"{self.tool_name}:{args_str}"

    def tool_signature(self) -> str:
        """Tool-only signature (ignores args)."""
        return self.tool_name

    def fuzzy_signature(self) -> str:
        """Normalized signature: tool + arg keys (ignores values)."""
        arg_keys = sorted(self.args.keys()) if isinstance(self.args, dict) else []
        return f"{self.tool_name}:{','.join(arg_keys)}"


class LoopDetector:
    def __init__(self, window_size: int = 10):
        self.window: deque[AgentAction] = deque(maxlen=window_size)
        self._signatures: dict[str, int] = {}  # signature → count in window

    def observe(self, action: AgentAction) -> dict:
        """Record an action and check for loops."""
        self.window.append(action)

        # Update signature counts
        sig = action.signature()
        self._signatures[sig] = self._signatures.get(sig, 0) + 1

        # Check all loop types
        checks = {
            "exact_repeat": self._check_exact_repeat(),
            "argument_cycle": self._check_argument_cycle(),
            "sequence_cycle": self._check_sequence_cycle(),
            "near_repeat": self._check_near_repeat(),
        }

        detected = {k: v for k, v in checks.items() if v["detected"]}

        return {
            "loop_detected": bool(detected),
            "loops": detected,
            "turns_observed": len(self.window),
        }

    def _check_exact_repeat(self) -> dict:
        """Same tool + same args called 3+ times."""
        for sig, count in self._signatures.items():
            if count >= 3:
                return {"detected": True, "type": "exact_repeat",
                        "signature": sig, "count": count,
                        "severity": "high" if count >= 4 else "medium"}
        return {"detected": False}

    def _check_argument_cycle(self) -> dict:
        """Same tool, args alternate between values (A-B-A-B pattern)."""
        if len(self.window) < 4:
            return {"detected": False}

        actions = list(self.window)
        recent = actions[-4:]

        # Check A-B-A-B pattern (same tool, alternating args)
        tools_same = (recent[0].tool_signature() == recent[2].tool_signature()
                      == recent[1].tool_signature() == recent[3].tool_signature())

        if not tools_same:
            return {"detected": False}

        sig_01 = recent[0].signature() == recent[2].signature()
        sig_02 = recent[1].signature() == recent[3].signature()
        different = recent[0].signature() != recent[1].signature()

        if sig_01 and sig_02 and different:
            return {"detected": True, "type": "argument_cycle",
                    "pattern": "A-B-A-B",
                    "tool": recent[0].tool_name,
                    "severity": "high"}

        return {"detected": False}

    def _check_sequence_cycle(self) -> dict:
        """Same sequence of tools repeats (e.g., search-read-write-search-read-write)."""
        if len(self.window) < 6:
            return {"detected": False}

        actions = list(self.window)

        # Try pattern lengths 2, 3
        for pattern_len in [2, 3]:
            if len(actions) >= pattern_len * 2:
                pattern = [a.tool_signature() for a in actions[-pattern_len*2:-pattern_len]]
                repeat = [a.tool_signature() for a in actions[-pattern_len:]]

                if pattern == repeat:
                    return {"detected": True, "type": "sequence_cycle",
                            "pattern_length": pattern_len,
                            "sequence": pattern,
                            "severity": "medium"}

        return {"detected": False}

    def _check_near_repeat(self) -> dict:
        """Same tool called within 3 turns with >80% arg similarity."""
        if len(self.window) < 2:
            return {"detected": False}

        actions = list(self.window)
        current = actions[-1]

        for i, prev in enumerate(actions[-4:-1]):  # Check last 3 actions
            if prev.tool_signature() != current.tool_signature():
                continue

            similarity = self._arg_similarity(prev.args, current.args)
            if similarity > 0.8:
                return {"detected": True, "type": "near_repeat",
                        "tool": current.tool_name,
                        "similarity": round(similarity, 2),
                        "turns_apart": i + 1,
                        "severity": "low" if similarity < 0.9 else "medium"}

        return {"detected": False}

    def _arg_similarity(self, args_a: dict, args_b: dict) -> float:
        """Jaccard similarity of arg values."""
        if not args_a and not args_b:
            return 1.0

        set_a = set(str(v) for v in args_a.values())
        set_b = set(str(v) for v in args_b.values())

        intersection = set_a & set_b
        union = set_a | set_b

        return len(intersection) / len(union) if union else 1.0

Step 2: Break Strategies

from enum import Enum

class BreakStrategy(Enum):
    WARN = "warn"              # Add warning to next message
    REDIRECT = "redirect"      # Inject "try a different approach"
    SKIP = "skip"              # Skip the tool call, return cached result
    FORCE_CONTEXT = "context"  # Force add context about the loop
    TERMINATE = "terminate"    # Stop the agent loop


class LoopBreaker:
    def __init__(self, detector: LoopDetector):
        self.detector = detector
        self._break_counts: dict[str, int] = {}  # loop_type → times broken
        self._cache: dict[str, str] = {}  # signature → last result

    def check_and_break(self, action: AgentAction,
                        max_breaks: int = 3) -> dict:
        result = self.detector.observe(action)

        if not result["loop_detected"]:
            return {"action": "allow", "loop_info": result}

        # Find the most severe loop
        loops = result["loops"]
        worst = max(loops.values(), key=lambda x: {
            "high": 3, "medium": 2, "low": 1
        }.get(x.get("severity", "low"), 0))

        loop_type = worst["type"]
        self._break_counts[loop_type] = self._break_counts.get(loop_type, 0) + 1

        # Escalating strategy
        breaks_so_far = self._break_counts[loop_type]

        if breaks_so_far >= max_breaks:
            strategy = BreakStrategy.TERMINATE
            message = (f"Agent has looped on {loop_type} "
                      f"{breaks_so_far} times. Terminating.")
        elif breaks_so_far >= 2:
            strategy = BreakStrategy.FORCE_CONTEXT
            message = (f"LOOP DETECTED ({loop_type}). "
                      f"You have tried this approach {breaks_so_far} times. "
                      f"You MUST try a fundamentally different approach. "
                      f"Previous attempts failed.")
        elif breaks_so_far >= 1:
            strategy = BreakStrategy.REDIRECT
            message = (f"Notice: you are repeating the same action ({loop_type}). "
                      f"Consider trying a different approach or tool.")
        else:
            strategy = BreakStrategy.WARN
            message = f"Warning: possible loop detected ({loop_type})."

        # Cache result for skip strategy
        if strategy == BreakStrategy.SKIP:
            cached = self._cache.get(action.signature())
            if cached:
                return {"action": "skip", "cached_result": cached,
                        "loop_info": worst, "strategy": strategy.value}

        return {
            "action": strategy.value,
            "message": message,
            "loop_info": worst,
            "break_count": breaks_so_far,
        }

The Complete Loop Guard

# Setup
detector = LoopDetector(window_size=10)
breaker = LoopBreaker(detector)

# Simulate agent actions
actions = [
    AgentAction("search_web", {"query": "python auth error"}),
    AgentAction("read_file", {"path": "auth.py"}),
    AgentAction("search_web", {"query": "python auth error"}),  # repeat
    AgentAction("read_file", {"path": "auth.py"}),              # repeat
    AgentAction("search_web", {"query": "python auth error"}),  # cycle!
    AgentAction("read_file", {"path": "auth.py"}),
    AgentAction("search_web", {"query": "python auth error"}),  # cycle!
]

for i, action in enumerate(actions):
    decision = breaker.check_and_break(action, max_breaks=3)
    status = decision["action"]

    if status == "terminate":
        print(f"Turn {i+1}: TERMINATED - {decision['message']}")
        break
    elif status == "allow":
        print(f"Turn {i+1}: OK - {action.tool_name}({list(action.args.values())})")
    else:
        print(f"Turn {i+1}: {status.upper()} - {decision['message'][:80]}")

Conclusion

Loop detection is the cheapest cost optimization in agent observability. A 200-line detector saves thousands of tokens per stuck session. The escalating break strategy (warn → redirect → context injection → terminate) gives the agent chances to self-correct before you pull the plug.

The key insight: exact repeats are easy to detect but rare. The dangerous loops are argument cycles (A-B-A-B) and sequence cycles (search-read-write repeating) — they look like progress but accomplish nothing. Check for all four pattern types every turn.
