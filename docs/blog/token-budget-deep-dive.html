Token Budget Management for AI Agents: Track, Predict, and Cut Costs

AI Agents Cost Control Python MCP

Your agent ran 90 iterations instead of 10. Each turn consumed 3,000 tokens. At $5/1M tokens, that one session cost $1.35. Do that 100 times a day, and you're burning $4,000/month on a single agent. Token budget management catches this before the bill arrives.

Step 1: Per-Turn Token Tracker

import sqlite3
import time
from dataclasses import dataclass, field

@dataclass
class TurnRecord:
    session_id: str
    turn_number: int
    model: str
    input_tokens: int
    output_tokens: int
    cache_read_tokens: int = 0
    cache_write_tokens: int = 0
    tool_calls: int = 0
    tool_tokens: int = 0  # Tokens in tool results
    cost_usd: float = 0.0
    timestamp: float = field(default_factory=time.time)


# Per-model pricing (USD per 1M tokens)
PRICING = {
    "gpt-4o": {"input": 2.50, "output": 10.00, "cache_read": 1.25},
    "gpt-4o-mini": {"input": 0.15, "output": 0.60, "cache_read": 0.075},
    "claude-sonnet-4": {"input": 3.00, "output": 15.00, "cache_read": 0.30},
    "claude-haiku-3.5": {"input": 0.25, "output": 1.25, "cache_read": 0.03},
    "gemini-2.5-flash": {"input": 0.075, "output": 0.30, "cache_read": 0.01},
}


def compute_cost(model: str, input_tokens: int, output_tokens: int,
                 cache_read: int = 0) -> float:
    rates = PRICING.get(model, PRICING["gpt-4o"])
    return (
        input_tokens / 1_000_000 * rates["input"]
        + output_tokens / 1_000_000 * rates["output"]
        - cache_read / 1_000_000 * (rates["input"] - rates["cache_read"])
    )


class TokenTracker:
    def __init__(self, db_path: str = ":memory:"):
        self.conn = sqlite3.connect(db_path)
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS turns (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                session_id TEXT, turn_number INTEGER,
                model TEXT, input_tokens INTEGER, output_tokens INTEGER,
                cache_read_tokens INTEGER, cache_write_tokens INTEGER,
                tool_calls INTEGER, tool_tokens INTEGER,
                cost_usd REAL, timestamp REAL
            )
        """)
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS budgets (
                session_id TEXT PRIMARY KEY,
                max_tokens INTEGER, max_cost REAL,
                max_turns INTEGER, created_at REAL
            )
        """)
        self.conn.commit()
        self._current_turn: dict[str, int] = {}

    def record_turn(self, record: TurnRecord) -> dict:
        cost = compute_cost(record.model, record.input_tokens,
                           record.output_tokens, record.cache_read_tokens)
        record.cost_usd = cost

        self.conn.execute("""
            INSERT INTO turns (session_id, turn_number, model,
                input_tokens, output_tokens, cache_read_tokens,
                cache_write_tokens, tool_calls, tool_tokens,
                cost_usd, timestamp)
            VALUES (?,?,?,?,?,?,?,?,?,?,?)
        """, (record.session_id, record.turn_number, record.model,
              record.input_tokens, record.output_tokens,
              record.cache_read_tokens, record.cache_write_tokens,
              record.tool_calls, record.tool_tokens,
              cost, record.timestamp))
        self.conn.commit()

        return self.check_budget(record.session_id)

    def check_budget(self, session_id: str) -> dict:
        budget = self.conn.execute(
            "SELECT max_tokens, max_cost, max_turns FROM budgets WHERE session_id = ?",
            (session_id,)).fetchone()

        if not budget:
            return {"status": "ok", "budget_set": False}

        max_tokens, max_cost, max_turns = budget

        totals = self.conn.execute("""
            SELECT COALESCE(SUM(input_tokens + output_tokens), 0),
                   COALESCE(SUM(cost_usd), 0),
                   COALESCE(MAX(turn_number), 0)
            FROM turns WHERE session_id = ?
        """, (session_id,)).fetchone()

        total_tokens, total_cost, total_turns = totals

        alerts = []
        if max_tokens and total_tokens >= max_tokens:
            alerts.append(f"TOKEN_BUDGET_EXCEEDED: {total_tokens}/{max_tokens}")
        if max_cost and total_cost >= max_cost:
            alerts.append(f"COST_BUDGET_EXCEEDED: ${total_cost:.4f}/${max_cost:.4f}")
        if max_turns and total_turns >= max_turns:
            alerts.append(f"TURN_BUDGET_EXCEEDED: {total_turns}/{max_turns}")

        return {
            "status": "exceeded" if alerts else "ok",
            "total_tokens": total_tokens,
            "total_cost": round(total_cost, 6),
            "total_turns": total_turns,
            "alerts": alerts,
        }

    def set_budget(self, session_id: str,
                   max_tokens: int | None = None,
                   max_cost: float | None = None,
                   max_turns: int | None = None) -> None:
        self.conn.execute("""
            INSERT OR REPLACE INTO budgets
            (session_id, max_tokens, max_cost, max_turns, created_at)
            VALUES (?,?,?,?,?)
        """, (session_id, max_tokens, max_cost, max_turns, time.time()))
        self.conn.commit()

    def session_summary(self, session_id: str) -> dict:
        row = self.conn.execute("""
            SELECT COUNT(*), COALESCE(SUM(input_tokens),0),
                   COALESCE(SUM(output_tokens),0),
                   COALESCE(SUM(cost_usd),0),
                   COALESCE(SUM(tool_tokens),0)
            FROM turns WHERE session_id = ?
        """, (session_id,)).fetchone()

        turns, inp, out, cost, tool_tok = row
        total = inp + out + tool_tok

        return {
            "turns": turns,
            "input_tokens": inp,
            "output_tokens": out,
            "tool_tokens": tool_tok,
            "total_tokens": total,
            "total_cost": round(cost, 6),
            "avg_tokens_per_turn": round(total / max(turns, 1)),
            "avg_cost_per_turn": round(cost / max(turns, 1), 6),
        }

Step 2: Predictive Cost Projection

class CostProjector:
    """Predict future costs based on historical patterns."""

    def __init__(self, tracker: TokenTracker):
        self.tracker = tracker

    def project_session_cost(self, session_id: str,
                             remaining_turns: int) -> dict:
        """Project total session cost based on avg so far."""
        summary = self.tracker.session_summary(session_id)

        if summary["turns"] == 0:
            return {"projected_cost": 0, "confidence": "none"}

        avg_cost = summary["total_cost"] / summary["turns"]
        avg_tokens = summary["avg_tokens_per_turn"]

        projected_additional = avg_cost * remaining_turns
        projected_total = summary["total_cost"] + projected_additional

        # Confidence based on sample size
        if summary["turns"] >= 20:
            confidence = "high"
        elif summary["turns"] >= 10:
            confidence = "medium"
        else:
            confidence = "low"

        return {
            "spent_so_far": round(summary["total_cost"], 4),
            "projected_additional": round(projected_additional, 4),
            "projected_total": round(projected_total, 4),
            "avg_cost_per_turn": round(avg_cost, 6),
            "avg_tokens_per_turn": avg_tokens,
            "remaining_turns": remaining_turns,
            "confidence": confidence,
        }

    def project_daily_cost(self, session_id: str,
                           sessions_per_day: int = 10) -> dict:
        """Project daily spend based on per-session averages."""
        summary = self.tracker.session_summary(session_id)

        if summary["turns"] == 0:
            return {"daily_projected": 0}

        avg_session_cost = summary["total_cost"]
        daily = avg_session_cost * sessions_per_day
        monthly = daily * 30

        return {
            "avg_session_cost": round(avg_session_cost, 4),
            "sessions_per_day": sessions_per_day,
            "daily_projected": round(daily, 2),
            "monthly_projected": round(monthly, 2),
        }

Step 3: Tool-Level Token Accounting

class ToolTokenAccountant:
    """Track which tools consume the most context."""

    def __init__(self, tracker: TokenTracker):
        self.tracker = tracker

    def record_tool_usage(self, session_id: str, turn: int,
                          tool_name: str, result_tokens: int,
                          args_tokens: int) -> None:
        """Record per-tool token consumption."""
        self.tracker.conn.execute("""
            INSERT INTO tool_usage
            (session_id, turn, tool_name, result_tokens, args_tokens, timestamp)
            VALUES (?,?,?,?,?,?)
        """, (session_id, turn, tool_name, result_tokens,
              args_tokens, time.time()))
        self.tracker.conn.commit()

    def tool_rankings(self, session_id: str | None = None) -> list[dict]:
        """Rank tools by total token consumption."""
        if session_id:
            rows = self.tracker.conn.execute("""
                SELECT tool_name,
                       COUNT(*) as calls,
                       SUM(result_tokens + args_tokens) as total_tokens,
                       AVG(result_tokens) as avg_result,
                       MAX(result_tokens) as max_result
                FROM tool_usage WHERE session_id = ?
                GROUP BY tool_name ORDER BY total_tokens DESC
            """, (session_id,)).fetchall()
        else:
            rows = self.tracker.conn.execute("""
                SELECT tool_name,
                       COUNT(*) as calls,
                       SUM(result_tokens + args_tokens) as total_tokens,
                       AVG(result_tokens) as avg_result,
                       MAX(result_tokens) as max_result
                FROM tool_usage GROUP BY tool_name
                ORDER BY total_tokens DESC
            """).fetchall()

        return [{"tool": r[0], "calls": r[1], "total_tokens": r[2],
                 "avg_result": round(r[3], 1), "max_result": r[4]}
                for r in rows]

    def optimize_toolset(self, session_id: str,
                         token_budget: int) -> list[str]:
        """Suggest which tools to keep based on usage vs token cost."""
        rankings = self.tool_rankings(session_id)

        # Knapsack: maximize calls within budget
        kept = []
        used = 0
        for tool in sorted(rankings, key=lambda x: x["calls"], reverse=True):
            avg_cost = tool["avg_result"]
            if used + avg_cost <= token_budget / 10:  # 10% of budget per tool
                kept.append(tool["tool"])
                used += avg_cost

        return kept

Optimization Strategies

StrategyTokens SavedImplementation
Compress conversation history40-60%Summarize turns 1-N after turn N+5
Trim tool schemas20-30%Only include relevant tools per turn
Cache system prompt50% input reductionKeep system prompt byte-stable
Cap tool result length10-20%Truncate results to 2000 tokens
Switch to cheaper model for easy tasks60-80% costRoute based on task complexity
Cache savings are real: With prompt caching, a 5,000-token system prompt cached across 20 turns saves 100,000 input tokens. At GPT-4o rates, that's $0.25 saved per session — just from keeping the prompt stable.

Common Pitfalls

Pitfall 1: Counting tokens after the API call. By then, you've already paid. Set budgets before the call and check them between turns.
Pitfall 2: Ignoring tool result tokens. Tool results go into the conversation history and get re-sent every turn. A 3,000-token file read is paid for on every subsequent turn.
Pitfall 3: Not tracking per-model costs. GPT-4o is 16x more expensive than GPT-4o-mini per output token. If a task works on mini, don't use 4o.

Conclusion

Token budget management is the difference between an agent that costs $50/month and one that costs $5,000/month for the same output quality. The system above is 250 lines of Python, uses SQLite for persistence, and catches runaway spending before it happens.

Three numbers to watch: avg cost per turn, avg tokens per turn, and tool result token contribution. If any of these trend upward across sessions, investigate before the bill spikes.
