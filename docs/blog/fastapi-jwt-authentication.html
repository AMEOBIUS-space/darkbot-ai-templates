1. JWT Token Generation

import jwt
from datetime import datetime, timedelta

def make_token(user_id: int) -> str:
    return jwt.encode(
        {"sub": str(user_id), "exp": datetime.utcnow() + timedelta(days=7)},
        SECRET, algorithm="HS256"
    )

3. Protected Routes

security = HTTPBearer()
def get_current_user(cred: HTTPAuthorizationCredentials = Depends(security)):
    payload = jwt.decode(cred.credentials, SECRET, algorithms=["HS256"])
    return int(payload["sub"])

@app.get("/items")
def list_items(uid: int = Depends(get_current_user)):
    return db.execute("SELECT * FROM items WHERE user_id=?", (uid,)).fetchall()

4. WebSocket Auth

@app.websocket("/ws/{token}")
async def ws(ws: WebSocket, token: str):
    payload = jwt.decode(token, SECRET, algorithms=["HS256"])
    await ws.accept()
    while True:
        data = await ws.receive_text()
        await ws.send_text(f"Echo: {data}")
