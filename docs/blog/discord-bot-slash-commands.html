Discord Bot with Slash Commands

Build a Discord bot with slash commands, polls, music queue, and moderation.

1. Setup

pip install discord.py
export DISCORD_TOKEN=your_token

2. Slash Commands

import discord
from discord.ext import commands
from discord import app_commands

bot = commands.Bot(command_prefix="!", intents=discord.Intents.all())

@bot.tree.command(name="ping")
async def ping(interaction: discord.Interaction):
 await interaction.response.send_message(f"Pong! {round(bot.latency*1000)}ms")

3. Polls with Reactions

@bot.tree.command(name="poll")
async def poll(interaction, question: str, *options: str):
 emojis = ["1⃣","2⃣","3⃣","4⃣"]
 embed = discord.Embed(title=f" {question}")
 for i, opt in enumerate(options):
 embed.add_field(name=emojis[i], value=opt)
 await interaction.response.send_message(embed=embed)
 msg = await interaction.original_response()
 for i in range(len(options)):
 await msg.add_reaction(emojis[i])

4. Auto-sync on Ready

@bot.event
async def on_ready():
 synced = await bot.tree.sync()
 print(f"Synced {len(synced)} commands")
