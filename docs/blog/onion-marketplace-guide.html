listing = mp.add_listing("Product", "desc", 0.5, "XMR", "vendor", "category")

2. Escrow Order

order = mp.place_order(listing.id, "buyer")
# Status: pending → escrow → shipped → completed

3. Payment Confirmation

mp.confirm_payment(order.id, "tx_hash")
# Releases escrow to vendor
