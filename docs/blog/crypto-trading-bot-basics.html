    async with aiohttp.ClientSession() as s:
        async with s.get(f"https://api.binance.com/api/v3/ticker/price?symbol={symbol}") as r:
            return float((await r.json())["price"])

2. RSI

def rsi(prices, period=14):
    gains = [max(prices[i]-prices[i-1],0) for i in range(1,len(prices))]
    losses = [max(-prices[i]+prices[i-1],0) for i in range(1,len(prices))]
    return 100 - (100 / (1 + sum(gains[-period:])/max(sum(losses[-period:]),0.01)))
