DebuggingMCPPython

2026-07-09

An agent made 23 tool calls, consumed 40,000 tokens, and produced the wrong answer. Which call went wrong? Without a trace, you are reading 23 tool results hoping to spot the moment the agent went off track.

>

Agent tracing solves this. Every tool call, every model response, every reasoning step is recorded in a structured trace that you can replay, filter, and analyze.

>

What a Trace Captures

>>>>>>>>
EventRecorded dataWhy it matters
Model callInput tokens, output tokens, model, latencyCost tracking, latency analysis
Tool callName, arguments, result, durationWhat the agent actually did
Decision pointAvailable tools, chosen tool, reasoningWhy the agent chose this path
ErrorException, stack trace, recovery actionFailure analysis
User messageText, timestampContext for the trace

The Trace Recorder

>
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
import json, time, hashlib

class EventType(Enum):
 MODEL_CALL = "model_call"
 TOOL_CALL = "tool_call"
 TOOL_RESULT = "tool_result"
 DECISION = "decision"
 ERROR = "error"
 USER_INPUT = "user_input"
 AGENT_RESPONSE = "agent_response"

@dataclass
class TraceEvent:
 type: EventType
 data: dict
 timestamp: str = ""
 elapsed_ms: int = 0
 event_id: str = ""
 
 def __post_init__(self):
 if not self.timestamp:
 self.timestamp = datetime.now().isoformat()
 if not self.event_id:
 raw = f"{self.timestamp}:{self.type.value}:{json.dumps(self.data, sort_keys=True)}"
 self.event_id = hashlib.md5(raw.encode()).hexdigest()[:8]

@dataclass
class AgentTracer:
 session_id: str = ""
 events: list = field(default_factory=list)
 _start_time: float = 0
 
 def __post_init__(self):
 self._start_time = time.time()
 
 def trace_model_call(self, model: str, input_tokens: int, 
 output_tokens: int, messages_count: int):
 self.events.append(TraceEvent(
 type=EventType.MODEL_CALL,
 data={
 "model": model,
 "input_tokens": input_tokens,
 "output_tokens": output_tokens,
 "messages": messages_count,
 "cost_estimate": input_tokens / 1000 * 0.003 + output_tokens / 1000 * 0.015
 },
 elapsed_ms=int((time.time() - self._start_time) * 1000)
 ))
 
 def trace_tool_call(self, name: str, args: dict):
 self.events.append(TraceEvent(
 type=EventType.TOOL_CALL,
 data={"name": name, "args": args},
 elapsed_ms=int((time.time() - self._start_time) * 1000)
 ))
 
 def trace_tool_result(self, name: str, result: str, duration_ms: int, error: str = ""):
 self.events.append(TraceEvent(
 type=EventType.TOOL_RESULT,
 data={
 "name": name,
 "result_preview": result[:200],
 "result_length": len(result),
 "duration_ms": duration_ms,
 "error": error
 }
 ))
 
 def trace_decision(self, available_tools: list, chosen: str, reason: str = ""):
 self.events.append(TraceEvent(
 type=EventType.DECISION,
 data={
 "available": available_tools,
 "chosen": chosen,
 "reason": reason,
 "options_count": len(available_tools)
 }
 ))
 
 def trace_error(self, error: str, context: dict = None):
 self.events.append(TraceEvent(
 type=EventType.ERROR,
 data={"error": error, "context": context or {}}
 ))

def render_trace(tracer: AgentTracer) ->str:
 """Render trace as readable text output."""
 lines = []
 lines.append(f"=== TRACE: {tracer.session_id} ({len(tracer.events)} events) ===")
 
 total_input = 0
 total_output = 0
 total_cost = 0
 tool_calls = {}
 errors = []
 
 for event in tracer.events:
 t = event.timestamp[11:19] # HH:MM:SS
 
 if event.type == EventType.MODEL_CALL:
 d = event.data
 total_input += d["input_tokens"]
 total_output += d["output_tokens"]
 total_cost += d["cost_estimate"]
 lines.append(f" [{t}] MODEL: {d['model']} "
 f"in={d['input_tokens']} out={d['output_tokens']} "
 f"(${d['cost_estimate']:.4f})")
 
 elif event.type == EventType.TOOL_CALL:
 name = event.data["name"]
 tool_calls[name] = tool_calls.get(name, 0) + 1
 args_preview = str(event.data["args"])[:60]
 lines.append(f" [{t}] CALL: {name}({args_preview})")
 
 elif event.type == EventType.TOOL_RESULT:
 d = event.data
 status = "ERROR" if d.get("error") else "OK"
 lines.append(f" [{t}] RESULT: {d['name']} {status} "
 f"({d['duration_ms']}ms, {d['result_length']} chars)")
 if d.get("error"):
 errors.append(d["error"])
 
 elif event.type == EventType.DECISION:
 d = event.data
 lines.append(f" [{t}] CHOSE: {d['chosen']} "
 f"({d['options_count']} options)")
 
 elif event.type == EventType.ERROR:
 lines.append(f" [{t}] ERROR: {event.data['error']}")
 errors.append(event.data["error"])
 
 # Summary
 lines.append("")
 lines.append("=== SUMMARY ===")
 lines.append(f" Tokens: {total_input:,} in + {total_output:,} out")
 lines.append(f" Cost: ${total_cost:.4f}")
 lines.append(f" Tools: {len(tool_calls)} unique, {sum(tool_calls.values())} calls")
 for name, count in sorted(tool_calls.items(), key=lambda x: -x[1]):
 lines.append(f" {name}: {count}x")
 if errors:
 lines.append(f" Errors: {len(errors)}")
 
 return "\n".join(lines)

import sqlite3

class PersistentTracer(AgentTracer):
 def __init__(self, db_path: str = "traces.db", session_id: str = ""):
 super().__init__(session_id=session_id)
 self.db = sqlite3.connect(db_path)
 self.db.execute("""
 CREATE TABLE IF NOT EXISTS traces (
 session_id TEXT, event_id TEXT, type TEXT,
 timestamp TEXT, elapsed_ms INTEGER, data TEXT
 )
 """)
 self.db.commit()
 
 def save(self):
 for event in self.events:
 self.db.execute(
 "INSERT INTO traces VALUES (?,?,?,?,?,?)",
 (self.session_id, event.event_id, event.type.value,
 event.timestamp, event.elapsed_ms,
 json.dumps(event.data))
 )
 self.db.commit()
 
 def load_session(self, session_id: str) ->list:
 rows = self.db.execute(
 "SELECT type, timestamp, elapsed_ms, data "
 "FROM traces WHERE session_id = ? ORDER BY elapsed_ms",
 (session_id,)
 ).fetchall()
 return [{"type": r[0], "time": r[1], 
 "elapsed": r[2], "data": json.loads(r[3])} for r in rows]

Pattern Analysis

Once you have traces from multiple sessions, analyze common patterns:

>
def find_inefficiencies(traces: list) ->list:
 """Find common inefficiencies across traces."""
 issues = []
 
 for trace in traces:
 events = trace.events
 
 # Check: same tool called 3+ times with same args
 seen = {}
 for e in events:
 if e.type == EventType.TOOL_CALL:
 key = f"{e.data['name']}:{str(e.data['args'])[:50]}"
 seen[key] = seen.get(key, 0) + 1
 if seen[key] >= 3:
 issues.append({
 "type": "repeat_calls",
 "trace": trace.session_id,
 "detail": f"{e.data['name']} called {seen[key]}x identically"
 })
 
 # Check: tool result over 5000 chars (context waste)
 for e in events:
 if e.type == EventType.TOOL_RESULT:
 if e.data.get("result_length", 0) >5000:
 issues.append({
 "type": "large_result",
 "trace": trace.session_id,
 "detail": f"{e.data['name']} returned {e.data['result_length']:,} chars"
 })
 
 # Check: error followed by same call (no recovery)
 for i in range(len(events) - 1):
 if (events[i].type == EventType.ERROR and 
 events[i+1].type == EventType.TOOL_CALL):
 issues.append({
 "type": "no_error_recovery",
 "trace": trace.session_id,
 "detail": f"Error then immediate retry without strategy change"
 })
 
 return issues
>
Run pattern analysis weekly. Common findings: agents reading the same file repeatedly, terminal outputs that are too large, and retries after errors that should trigger a different approach.
