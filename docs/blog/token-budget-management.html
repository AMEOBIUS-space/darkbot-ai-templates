Token Budget Management: Track and Limit Agent Costs

MCP Cost Python

2026-07-09

An agent that runs 40 tool calls in a loop does not just waste time — it wastes money. At $0.003 per 1K input tokens, a single runaway loop can cost $0.50 in five minutes. At scale, this adds up fast.

This article shows a token budget system that tracks usage per conversation, per tool, and per turn — and shuts down runaway agents before they bankrupt you.

The Token Tracker

from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum

class BudgetAction(Enum):
    CONTINUE = "continue"
    WARN = "warn"
    STOP = "stop"

@dataclass
class TurnUsage:
    turn_number: int
    input_tokens: int = 0
    output_tokens: int = 0
    tool_name: str = ""
    timestamp: str = ""

@dataclass
class TokenBudgetManager:
    max_total_tokens: int = 50000
    max_per_turn: int = 8000
    max_per_tool: int = 3000
    warn_threshold: float = 0.7  # warn at 70% of budget
    stop_threshold: float = 0.95  # stop at 95%
    
    turns: list = field(default_factory=list)
    tool_usage: dict = field(default_factory=dict)  # tool_name -> total tokens
    
    @property
    def total_input(self) -> int:
        return sum(t.input_tokens for t in self.turns)
    
    @property
    def total_output(self) -> int:
        return sum(t.output_tokens for t in self.turns)
    
    @property
    def total_tokens(self) -> int:
        return self.total_input + self.total_output
    
    @property
    def turn_count(self) -> int:
        return len(self.turns)
    
    def record(self, input_tokens: int, output_tokens: int, tool_name: str = "") -> BudgetAction:
        """Record usage for a turn. Returns recommended action."""
        turn = TurnUsage(
            turn_number=len(self.turns) + 1,
            input_tokens=input_tokens,
            output_tokens=output_tokens,
            tool_name=tool_name,
            timestamp=datetime.now().isoformat()
        )
        self.turns.append(turn)
        
        if tool_name:
            self.tool_usage[tool_name] = \
                self.tool_usage.get(tool_name, 0) + input_tokens + output_tokens
        
        return self.check_budget()
    
    def check_budget(self) -> BudgetAction:
        """Check if any budget threshold is exceeded."""
        # Total budget check
        ratio = self.total_tokens / self.max_total_tokens
        if ratio >= self.stop_threshold:
            return BudgetAction.STOP
        if ratio >= self.warn_threshold:
            return BudgetAction.WARN
        
        # Per-turn check (last turn)
        if self.turns:
            last = self.turns[-1]
            last_total = last.input_tokens + last.output_tokens
            if last_total >= self.max_per_turn:
                return BudgetAction.WARN
        
        # Per-tool check
        for tool, usage in self.tool_usage.items():
            if usage >= self.max_per_tool:
                return BudgetAction.WARN
        
        return BudgetAction.CONTINUE
    
    def summary(self) -> dict:
        """Get usage summary."""
        return {
            "total_tokens": self.total_tokens,
            "input_tokens": self.total_input,
            "output_tokens": self.total_output,
            "turns": self.turn_count,
            "avg_per_turn": self.total_tokens // max(1, self.turn_count),
            "budget_used_pct": round(
                self.total_tokens / self.max_total_tokens * 100, 1
            ),
            "top_tools": sorted(
                self.tool_usage.items(), 
                key=lambda x: x[1], 
                reverse=True
            )[:5]
        }
    
    def cost_estimate(self, input_rate: float = 0.003, output_rate: float = 0.015) -> float:
        """Estimate cost in USD. Rates per 1K tokens."""
        return (self.total_input / 1000 * input_rate + 
                self.total_output / 1000 * output_rate)

Integration with Agent Loop

# In your agent's main loop:
budget = TokenBudgetManager(
    max_total_tokens=50000,
    warn_threshold=0.7,
    stop_threshold=0.95
)

for turn in range(max_turns):
    response = call_llm(messages, tools)
    
    # Record token usage
    action = budget.record(
        input_tokens=response.usage.prompt_tokens,
        output_tokens=response.usage.completion_tokens,
        tool_name=response.tool_calls[0].name if response.tool_calls else ""
    )
    
    if action == BudgetAction.STOP:
        # Inject budget exceeded message
        messages.append({
            "role": "system",
            "content": f"[BUDGET EXCEEDED] Token budget used: "
                       f"{budget.total_tokens}/{budget.max_total_tokens}. "
                       f"Provide your best answer with the information you have."
        })
        break
    
    if action == BudgetAction.WARN:
        s = budget.summary()
        messages.append({
            "role": "system",
            "content": f"[BUDGET WARNING] {s['budget_used_pct']}% used. "
                       f"Be concise. Top tool: {s['top_tools'][0][0]} "
                       f"({s['top_tools'][0][1]} tokens)."
        })
    
    # Process tool calls...
The STOP action is critical. Without it, a single runaway conversation can consume $5-10 in API costs. Set the threshold to 95% — the agent gets one final turn to summarize.

Persistent Budget Tracking

For multi-session agents, persist budget data to SQLite:

import sqlite3
from pathlib import Path

class PersistentBudget(TokenBudgetManager):
    def __init__(self, db_path: str = "budget.db", **kwargs):
        super().__init__(**kwargs)
        self.db = sqlite3.connect(db_path)
        self.db.execute("""
            CREATE TABLE IF NOT EXISTS usage (
                session_id TEXT,
                turn_number INTEGER,
                input_tokens INTEGER,
                output_tokens INTEGER,
                tool_name TEXT,
                timestamp TEXT,
                cost_usd REAL
            )
        """)
        self.db.commit()
    
    def record(self, input_tokens: int, output_tokens: int, 
               tool_name: str = "", session_id: str = "default") -> BudgetAction:
        action = super().record(input_tokens, output_tokens, tool_name)
        
        cost = (input_tokens / 1000 * 0.003 + output_tokens / 1000 * 0.015)
        self.db.execute(
            "INSERT INTO usage VALUES (?, ?, ?, ?, ?, ?, ?)",
            (session_id, len(self.turns), input_tokens, output_tokens,
             tool_name, datetime.now().isoformat(), cost)
        )
        self.db.commit()
        return action
    
    def session_summary(self, session_id: str = "default") -> dict:
        row = self.db.execute(
            "SELECT SUM(input_tokens), SUM(output_tokens), SUM(cost_usd), "
            "COUNT(*) FROM usage WHERE session_id = ?",
            (session_id,)
        ).fetchone()
        return {
            "input_tokens": row[0] or 0,
            "output_tokens": row[1] or 0,
            "total_cost_usd": round(row[2] or 0, 4),
            "turns": row[3] or 0
        }
Use persistent tracking to identify which tasks cost the most. This data helps you optimize prompt caching, tool selection, and context compression.
